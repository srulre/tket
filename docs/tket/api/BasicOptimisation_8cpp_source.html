45 // this method annihilates all primitives next to each other (accounting for
46 // previous annihilations)
47 // also removes redundant non-classically controlled Z basis gates before a z
48 // basis measurement so that eg. -H-X-X-H- always annihilates to -----
49 static bool redundancy_removal(Circuit &circ) {
50  bool success = false;
51  bool found_redundancy = true;
52  IndexMap im = circ.index_map();
53  std::set<IVertex> old_affected_verts;
54  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
55  old_affected_verts.insert({im.at(v), v});
56  }
57  VertexList bin;
58  while (found_redundancy) {
59  std::set<IVertex> new_affected_verts;
60  for (const IVertex &p : old_affected_verts) {
61  remove_redundancy(circ, p.second, bin, new_affected_verts, im);
62  }
63  found_redundancy = new_affected_verts.size() != 0;
64  success |= found_redundancy;
65  old_affected_verts = new_affected_verts;
66  }
67  circ.remove_vertices(
68  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
69  return success;
70 }

72 // called by the previous method. This should generally not be called
73 // independently
74 static bool remove_redundancy(
75  Circuit &circ, const Vertex &vert, VertexList &bin,
76  std::set<IVertex> &new_affected_verts, IndexMap &im) {
77  const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
78  const OpDesc desc = op->get_desc();
79  if (!desc.is_gate()) return false;
80  if (circ.n_out_edges(vert) == 0 || circ.n_in_edges(vert) == 0) {
81  return false; // either a boundary vert or we have already removed it
82  }
83 
84  auto remove_single_vertex = [&bin, &circ, &new_affected_verts,
85  &im](const Vertex &v_remove) {
86  bin.push_back(v_remove);
87  for (const Vertex &l : circ.get_predecessors(v_remove)) {
88  new_affected_verts.insert({im.at(l), l});
89  }
90  circ.remove_vertex(
91  v_remove, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
92  };
93  // remove 0 angle rotations from circuit
94  std::optional<double> a = op->is_identity();
95  if (a) {
96  remove_single_vertex(vert);
97  circ.add_phase(a.value());
98  return true;
99  } else if (desc.type() == OpType::noop) {
100  // remove "noop" gates from circuit
101  remove_single_vertex(vert);
102  return true;
103  }
104  VertexVec kids = circ.get_successors(vert);
105 
106  // if op is immediately followed by a z basis measurement on all qubits,
107  // remove
108  if (circ.n_out_edges_of_type(vert, EdgeType::Classical) == 0) {
109  bool z_followed_by_measures = true;
110  for (port_t port = 0; port < kids.size() && z_followed_by_measures;
111  port++) {
112  if (circ.get_OpType_from_Vertex(kids[port]) == OpType::Measure) {
113  z_followed_by_measures &= op->commutes_with_basis(Pauli::Z, port);
114  } else {
115  z_followed_by_measures = false;
116  }
117  }
118  if (z_followed_by_measures) {
119  remove_single_vertex(vert);
120  return true;
121  }
122  }
123 
124  // check that both the vertex and its successor have each other and only each
125  // other
126  if ((kids.size() == 1) && (circ.get_predecessors(kids[0]).size() == 1)) {
127  // check that the ports match up between vertices
128  Vertex b = kids[0];
129  EdgeVec ins = circ.get_in_edges(b);
130  for (const Edge &in : ins) {
131  if (circ.get_source_port(in) != circ.get_target_port(in)) return false;
132  }
133 
134  // check that the classical edges match up correctly
135  if (circ.n_in_edges_of_type(vert, EdgeType::Boolean) != 0) return false;
136 
137  const Op_ptr b_op = circ.get_Op_ptr_from_Vertex(b);
138  const OpDesc b_desc = b_op->get_desc();
139 
140  if (!b_desc.is_oneway()) {
141  // if A = B.dagger(), AB = I
142  // This method cannot detect matches between rotation gates.
143  // Rotation gates are covered by the rotation gate combiner, everything
144  // else in this.
145  if (*b_op->dagger() == *op) {
146  bin.push_back(vert);
147  bin.push_back(b);
148  VertexVec last_verts = circ.get_predecessors(vert);
149  for (const Vertex &l : last_verts) {
150  new_affected_verts.insert({im.at(l), l});
151  }
152  VertexList to_detach{vert, b};
153  // detached from circuit but not removed from graph
154  circ.remove_vertices(
155  to_detach, Circuit::GraphRewiring::Yes,
156  Circuit::VertexDeletion::No);
157  return true;
158  } else if (desc.is_rotation()) {
159  // combine two rotation gates together, then if the combined
160  // operation is the identity up to phase, remove from circuit
161  if (b_desc.type() == desc.type()) {
162  Expr expr1 = op->get_params()[0];
163  Expr expr2 = b_op->get_params()[0];
164  VertexVec last_verts = circ.get_predecessors(vert);
165  for (const Vertex &l : last_verts) {
166  new_affected_verts.insert({im.at(l), l});
167  }
168  circ.remove_vertex(
169  b, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
170  bin.push_back(b);
171  std::vector<Expr> params_new = {expr1 + expr2};
172  Op_ptr op_new = get_op_ptr(desc.type(), params_new, ins.size());
173  std::optional<double> a = op_new->is_identity();
174  if (a) {
175  bin.push_back(vert);
176  circ.remove_vertex(
177  vert, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
178  circ.add_phase(a.value());
179  } else {
180  new_affected_verts.insert({im[vert], vert});
181  circ.dag[vert].op = op_new;
182  }
183  return true;
184  }
185  }
186  }
187  }
188  return false;
189 }

195 // moves single qubit operations past multiqubit operations they commute with,
196 // towards front of circuit (hardcoded)
197 static bool commute_singles_to_front(Circuit &circ) {
198  bool success = false;
199  // follow each qubit path from output to input
200  for (const Qubit &q : circ.all_qubits()) {
201  Vertex prev_v = circ.get_out(q);
202  Edge current_e = circ.get_nth_in_edge(prev_v, 0);
203  Vertex current_v = circ.source(current_e);
204  while (!is_initial_q_type(circ.get_OpType_from_Vertex(current_v))) {
205  const Op_ptr curr_op = circ.get_Op_ptr_from_Vertex(current_v);
206  // if current vertex is a multiqubit gate
207  if (circ.n_in_edges_of_type(current_v, EdgeType::Quantum) > 1 &&
208  curr_op->get_desc().is_gate()) {
209  // need gate check to access commuting_basis
210  const std::pair<port_t, port_t> ports = circ.get_ports(current_e);
211  check_prev_commutes:
212  const Op_ptr prev_op = circ.get_Op_ptr_from_Vertex(prev_v);
213  // if previous vertex is single qubit gate
214  if (prev_op->get_desc().is_gate() &&
215  circ.n_in_edges_of_type(prev_v, EdgeType::Quantum) == 1) {
216  const std::optional<Pauli> prev_colour =
217  prev_op->commuting_basis(ports.second);
218 
219  if (curr_op->commutes_with_basis(prev_colour, ports.first)) {
220  // subsequent op on qubit path is a single qubit gate
221  // and commutes with current multi qubit gate
222  success = true;
223  circ.remove_vertex(
224  prev_v, Circuit::GraphRewiring::Yes,
225  Circuit::VertexDeletion::No);
226  Edge rewire_edge = circ.get_nth_in_edge(current_v, ports.first);
227  circ.rewire(prev_v, {rewire_edge}, {EdgeType::Quantum});
228  current_e = circ.get_nth_out_edge(current_v, ports.first);
229  prev_v = circ.target(current_e);
230  // check if new previous gate can be commuted through too
231  goto check_prev_commutes;
232  }
233  }
234  }
235  // move to next vertex (towards input)
236  prev_v = current_v;
237  std::tie(current_v, current_e) = circ.get_prev_pair(current_v, current_e);
238  }
239  }
240 
241  return success;
242 }

255 static bool replace_two_qubit_interaction(
256  Circuit &circ, Interaction &i, std::map<Qubit, Edge> &current_edges,
257  VertexList &bin, const double cx_fidelity = 1.) {
258  EdgeVec in_edges = {i.e0, i.e1};
259  EdgeVec out_edges = {current_edges[i.q0], current_edges[i.q1]};
260  Edge next0, next1;
261  bool q0_is_out = is_final_q_type(
262  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q0])));
263  bool q1_is_out = is_final_q_type(
264  circ.get_OpType_from_Vertex(circ.target(current_edges[i.q1])));
265  if (!q0_is_out) {
266  next0 = circ.get_next_edge(
267  circ.target(current_edges[i.q0]), current_edges[i.q0]);
268  }
269  if (!q1_is_out) {
270  next1 = circ.get_next_edge(
271  circ.target(current_edges[i.q1]), current_edges[i.q1]);
272  }
273  Subcircuit sub = {in_edges, out_edges, i.vertices};
274  Circuit subc = circ.subcircuit(sub);
275  Eigen::Matrix4cd mat = get_matrix_from_2qb_circ(subc);
276  Circuit replacement = two_qubit_canonical(mat);
277  // TODO: for now we decompose all the way to CX. In the future, this pass
278  // should output TK2, and decompose to CX (or other gates) later if necessary.
279  TwoQbFidelities fid;
280  fid.CX_fidelity = cx_fidelity;
281  (decompose_TK2(fid) >> squash_1qb_to_tk1()).apply(replacement);
282  const int nb_cx_old = subc.count_gates(OpType::CX);
283  const int nb_cx_new = replacement.count_gates(OpType::CX);
284  if (nb_cx_new < nb_cx_old) {
285  bin.insert(bin.end(), sub.verts.begin(), sub.verts.end());
286  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
287  if (!q0_is_out) {
288  current_edges[i.q0] = circ.get_last_edge(circ.source(next0), next0);
289  }
290  if (!q1_is_out) {
291  current_edges[i.q1] = circ.get_last_edge(circ.source(next1), next1);
292  }
293  return true;
294  } else {
295  return false;
296  }

299 Transform commute_and_combine_HQS2() {
300  return Transform([](Circuit &circ) {
301  bool success = false;
302  VertexList bin;
303  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
304  EdgeVec outs = circ.get_all_out_edges(v);
305  if (circ.get_OpType_from_Vertex(v) == OpType::ZZMax && outs.size() == 2) {
306  Vertex next0 = boost::target(outs[0], circ.dag);
307  Vertex next1 = boost::target(outs[1], circ.dag);
308  if (next0 == next1 &&
309  circ.get_OpType_from_Vertex(next0) == OpType::ZZMax) {
310  success = true;
311  EdgeVec h_in = circ.get_in_edges(v);
312  EdgeVec h_out = circ.get_all_out_edges(next0);
313  if (circ.get_target_port(outs[0]) != 0) {
314  h_out = {h_out[1], h_out[0]};
315  }
316  bin.push_back(v);
317  bin.push_back(next0);
318  Subcircuit sub = {h_in, h_out};
319  circ.substitute(
320  CircPool::two_Rz1(), sub, Circuit::VertexDeletion::No);
321  circ.add_phase(0.5);
322  continue;
323  }
324  if (circ.get_OpType_from_Vertex(next0) == OpType::Rz) {
325  success = true;
326  circ.remove_vertex(
327  next0, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
328  Edge in_0 = circ.get_nth_in_edge(v, 0);
329  circ.rewire(next0, {in_0}, {EdgeType::Quantum});
330  }
331  if (circ.get_OpType_from_Vertex(next1) == OpType::Rz) {
332  success = true;
333  circ.remove_vertex(
334  next1, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
335  Edge in_1 = circ.get_nth_in_edge(v, 1);
336  circ.rewire(next1, {in_1}, {EdgeType::Quantum});
337  }
338  }
339  }
340  circ.remove_vertices(
341  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
342  return success;
343  });
344 }

346 // TODO:: Work around classically controlled stuff
347 Transform two_qubit_squash(double cx_fidelity) {
348  return Transform([cx_fidelity](Circuit &circ) {
349  bool success = false;
350  VertexList bin;
351  // Get map from vertex/port to qubit number
352  std::map<VertPort, Qubit> v_to_qb;
353  std::map<Qubit, Edge> current_edge_on_qb;
354  std::vector<Interaction> i_vec;
355  std::map<Qubit, int> current_interaction;
356  for (const Qubit &qb : circ.all_qubits()) {
357  for (const VertPort &vp : circ.unit_path(qb)) {
358  v_to_qb.insert({vp, qb});
359  }
360  Vertex input = circ.get_in(qb);
361  Edge e = circ.get_nth_out_edge(input, 0);
362  current_edge_on_qb[qb] = e;
363  current_interaction[qb] = -1;
364  }
365  SliceVec slices = circ.get_slices();
366  slices.insert(slices.begin(), circ.q_inputs());
367  slices.push_back(circ.q_outputs());
368  for (SliceVec::iterator s = slices.begin(); s != slices.end(); ++s) {
369  for (Slice::iterator v = s->begin(); v != s->end(); ++v) {
370  const Op_ptr o = circ.get_Op_ptr_from_Vertex(*v);
371  OpType type = o->get_type();
372  unsigned n_ins = circ.n_in_edges_of_type(*v, EdgeType::Quantum);
373  // Measures, resets, outputs, barriers, symbolic gates, and many-qubit
374  // gates close interactions
375  if (is_projective_type(type) || is_final_q_type(type) ||
376  type == OpType::Barrier || n_ins > 2 ||
377  !o->free_symbols().empty()) {
378  for (port_t port = 0; port < n_ins; port++) {
379  Qubit q = v_to_qb.at({*v, port});
380  int i = current_interaction[q];
381  if (i != -1) {
382  if (i_vec[i].count >= 2) {
383  // Replace subcircuit
384  success |= replace_two_qubit_interaction(
385  circ, i_vec[i], current_edge_on_qb, bin, cx_fidelity);
386  }
387  current_interaction[i_vec[i].q0] = -1;
388  current_interaction[i_vec[i].q1] = -1;
389  }
390  if (!is_final_q_type(type)) {
391  current_edge_on_qb[q] =
392  circ.get_next_edge(*v, current_edge_on_qb[q]);
393  }
394  }
395  continue;
396  }
397 
398  // Check for 2qb gate
399  if (circ.n_in_edges_of_type(*v, EdgeType::Quantum) == 2) {
400  Qubit q0 = v_to_qb.at({*v, 0});
401  Qubit q1 = v_to_qb.at({*v, 1});
402  int i0 = current_interaction[q0];
403  int i1 = current_interaction[q1];
404  // If they are already interacting, extend it
405  if (i0 != -1 && i0 == i1) {
406  i_vec[i0].count++;
407  i_vec[i0].vertices.insert(*v);
408  current_edge_on_qb[q0] =
409  circ.get_next_edge(*v, current_edge_on_qb[q0]);
410  current_edge_on_qb[q1] =
411  circ.get_next_edge(*v, current_edge_on_qb[q1]);
412  continue;
413  }
414  // End any other interactions on q0
415  if (i0 != -1) {
416  if (i_vec[i0].count >= 2) {
417  // Replace subcircuit
418  success |= replace_two_qubit_interaction(
419  circ, i_vec[i0], current_edge_on_qb, bin, cx_fidelity);
420  }
421  current_interaction[i_vec[i0].q0] = -1;
422  current_interaction[i_vec[i0].q1] = -1;
423  }
424  // End any other interactions on q1
425  if (i1 != -1) {
426  if (i_vec[i1].count >= 2) {
427  // Replace subcircuit
428 
429  success |= replace_two_qubit_interaction(
430  circ, i_vec[i1], current_edge_on_qb, bin, cx_fidelity);
431  }
432  current_interaction[i_vec[i1].q0] = -1;
433  current_interaction[i_vec[i1].q1] = -1;
434  }
435  // Add new interaction
436  Interaction new_i(q0, q1);
437  new_i.e0 = current_edge_on_qb[q0];
438  new_i.e1 = current_edge_on_qb[q1];
439  new_i.count = 1;
440  new_i.vertices = {*v};
441  current_interaction[q0] = i_vec.size();
442  current_interaction[q1] = i_vec.size();
443  i_vec.push_back(new_i);
444  current_edge_on_qb[q0] =
445  circ.get_next_edge(*v, current_edge_on_qb[q0]);
446  current_edge_on_qb[q1] =
447  circ.get_next_edge(*v, current_edge_on_qb[q1]);
448  continue;
449  }
450 
451  // Otherwise, we don't care about other vertices, so just update edges
452  // and add vertices if interactions exist
453  for (port_t i = 0; i < circ.n_in_edges(*v); i++) {
454  Qubit q = v_to_qb.at({*v, i});
455  current_edge_on_qb[q] = circ.get_next_edge(*v, current_edge_on_qb[q]);
456  int inter = current_interaction[q];
457  if (inter != -1) {
458  i_vec[inter].vertices.insert(*v);
459  }
460  }
461  }
462  }
463  circ.remove_vertices(
464  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
465  return success;
466  });
467 }

469 // Given a 'SWAP_chain', finds edge in chain (or qubit wire) with best fidelity
470 // and rewires the associated single qubit vertex in to it
471 static bool find_edge_rewire_vertex(
472  Circuit &circ,
473  std::pair<std::vector<std::pair<Edge, double>>, Vertex> &entry) {
474  std::vector<std::pair<Edge, double>> candidates = entry.first;
475  std::pair<Edge, double> best_pair = candidates[0];
476  bool do_rewire = false; // prevents rewiring if current edge is best edge
477  // (causes many issues...)
478  for (unsigned i = 1; i < candidates.size(); i++) { // find best edge
479  if (candidates[i].second > best_pair.second) {
480  best_pair = candidates[i];
481  do_rewire = true;
482  }
483  }
484  if (do_rewire) {
485  // rewire in to best edge
486  circ.remove_vertex(
487  entry.second, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
488  circ.rewire(entry.second, {best_pair.first}, {EdgeType::Quantum});
489  }
490  return do_rewire;
491 }

493 // Given a SWAP vertex, which has some predecessor SWAP vertex, find the
494 // 'SWAP_chain' this predecessor SWAP vertex is in and add to it
495 static void extend_SWAP_chain(
496  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
497  &swap_chains,
498  Edge entry_edge, Node entry_node, const Edge &match, const Circuit &circ,
499  const DeviceCharacterisation &characterisation) {
500  for (auto it = swap_chains.begin(); it != swap_chains.end(); ++it) {
501  if ((*it).first.back().first == match) {
502  // extend chain, adding a new pair of edge and double to the end
503  (*it).first.push_back(
504  {entry_edge,
505  1.0 - characterisation.get_error(
506  entry_node, circ.get_OpType_from_Vertex((*it).second))});
507  return;
508  }
509  }
510 }

512 // Finds sequences of adjacent SWAP gates, with a predecessor Single Qubit
513 // Vertex. The error rate of the required Single Qubit Vertex is stored for each
514 // of the PhysicalQubits the LogicalQubit passes through Once 'SWAP_chains' are
515 // found throughout the whole circuit, predecessor Single Qubit Vertices are
516 // rewired into the edge with best error rate
517 static bool find_rewire_sq(
518  Circuit &circ, const DeviceCharacterisation &characterisation) {
519  std::list<std::pair<std::vector<std::pair<Edge, double>>, Vertex>>
520  swap_chains;
521  for (auto it = circ.begin(); it != circ.end(); ++it) {
522  if (it->get_op_ptr()->get_type() == OpType::SWAP) {
523  // find SWAP, if either predecessor is a single qubit unitary
524  // find resulting swap chain...
525  Vertex swap_vert = it.get_vertex();
526  unit_vector_t qubits = it->get_args();
527  node_vector_t nodes = {qubits.begin(), qubits.end()};
528  VertexVec pred_verts = circ.get_predecessors(swap_vert);
529  EdgeVec pred_edges = circ.get_in_edges(swap_vert);
530  EdgeVec post_edges = circ.get_all_out_edges(swap_vert);
531  for (unsigned i = 0; i < pred_verts.size(); i++) {
532  OpType optype = circ.get_OpType_from_Vertex(
533  pred_verts[i]); // OPT IS PREDECESSOR OPTYPE
534  if (circ.detect_singleq_unitary_op(pred_verts[i])) {
535  // wire has single qubit unitary->add a new swap chain
536  std::vector<std::pair<Edge, double>> swap_chain = {
537  {pred_edges[i],
538  1.0 - characterisation.get_error(nodes[i], optype)},
539  {post_edges[1 - i],

548  }

550  }

557  success = true;
558  }
559  swap_chains.erase(swap_chains.begin());
560  }

562 }
