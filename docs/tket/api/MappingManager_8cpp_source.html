32 bool MappingManager::route_circuit_with_maps(
33  Circuit& circuit, const std::vector<RoutingMethodPtr>& routing_methods,
34  std::shared_ptr<unit_bimaps_t> maps, bool label_isolated_qubits) const {
35  if (circuit.n_qubits() > this->architecture_->n_nodes()) {
36  std::string error_string =
37  "Circuit has" + std::to_string(circuit.n_qubits()) +
38  " logical qubits. Architecture has " +
39  std::to_string(this->architecture_->n_nodes()) +
40  " physical qubits. Circuit to be routed can not have more "
41  "qubits than the Architecture.";
42  throw MappingManagerError(error_string);
43  }
44 
45  // mapping_frontier tracks boundary between routed & un-routed in circuit
46  // when initialised, boundary is over output edges of input vertices
47  MappingFrontier_ptr mapping_frontier;
48  if (!maps->initial.empty() && !maps->final.empty()) {
49  mapping_frontier = std::make_shared<MappingFrontier>(circuit, maps);
50  } else {
51  mapping_frontier = std::make_shared<MappingFrontier>(circuit);
52  }
53  // updates routed/un-routed boundary
54 
55  mapping_frontier->advance_frontier_boundary(this->architecture_);
56 
57  auto check_finish = [&mapping_frontier]() {
58  for (const std::pair<UnitID, VertPort>& pair :
59  mapping_frontier->linear_boundary->get<TagKey>()) {
60  Edge e = mapping_frontier->circuit_.get_nth_out_edge(
61  pair.second.first, pair.second.second);
62  Vertex v = mapping_frontier->circuit_.target(e);
63  OpType ot = mapping_frontier->circuit_.get_OpType_from_Vertex(v);
64  if (!is_final_q_type(ot) && ot != OpType::ClOutput) {
65  return false;
66  }
67  }
68  return true;
69  };
70 
71  bool circuit_modified = !check_finish();
72  while (!check_finish()) {
73  // The order methods are passed in std::vector<RoutingMethod> is
74  // the order they are run
75  // If a method performs better but only on specific subcircuits,
76  // rank it earlier in the passed vector
77  bool valid_methods = false;
78  for (const auto& rm : routing_methods) {
79  // true => can use held routing method
80  std::pair<bool, unit_map_t> bool_map =
81  rm->routing_method(mapping_frontier, this->architecture_);
82  if (bool_map.first) {
83  valid_methods = true;
84  if (bool_map.second.size() > 0) {
85  std::map<Node, Node> node_map;
86  for (const auto& x : bool_map.second) {
87  node_map.insert({Node(x.first), Node(x.second)});
88  }
89  for (const std::pair<Node, Node>& swap :
90  BestTsaWithArch::get_swaps(*this->architecture_, node_map)) {
91  mapping_frontier->add_swap(swap.first, swap.second);
92  }
93  }
94  break;
95  }
96  }
97  if (!valid_methods) {
98  throw MappingManagerError(
99  "No RoutingMethod suitable to map given subcircuit.");
100  }
101  // find next routed/unrouted boundary given updates
102  mapping_frontier->advance_frontier_boundary(this->architecture_);
103  }
104  // there may still be some unlabelled qubits
105  if (label_isolated_qubits) {
106  circuit_modified = true;
107 
108  unit_map_t placement;
109  qubit_vector_t to_place;
110  std::vector<Node> placed;
111 
112  // Find which/if any qubits need placing
113  for (const Qubit& q : mapping_frontier->circuit_.all_qubits()) {
114  Node n(q);
115  if (!this->architecture_->node_exists(n)) {
116  // Ancilla qubits can be assigned during routing
117  // If some qubits are unplaced then its possible the returned circuit
118  // has more qubits than the architecture has nodes, which is bad instead
119  // at least assign any unlabelled qubits to any ancilla nodes to prevent
120  // this
121  if (mapping_frontier->ancilla_nodes_.size() > 0) {
122  circuit_modified = true;
123  Node ancilla = *mapping_frontier->ancilla_nodes_.begin();
124  mapping_frontier->merge_ancilla(q, ancilla);
125  mapping_frontier->ancilla_nodes_.erase(
126  mapping_frontier->ancilla_nodes_.begin());
127  placed.push_back(n);
128  } else {
129  to_place.push_back(n);
130  }
131  } else {
132  placed.push_back(n);
133  // if already placed, make sure qubit retains placement
134  placement.insert({n, n});
135  }
136  }
137  // avoid doing std::set_difference unless qubits need to be placed
138  unsigned n_placed = to_place.size();
139  if (n_placed > 0) {
140  std::vector<Node> difference,
141  architecture_nodes = this->architecture_->get_all_nodes_vec();
142  std::set_difference(
143  architecture_nodes.begin(), architecture_nodes.end(), placed.begin(),
144  placed.end(), std::inserter(difference, difference.begin()));
145  // should always be enough remaining qubits to assign unplaced qubits to
146  TKET_ASSERT(difference.size() >= n_placed);
147  for (unsigned i = 0; i < n_placed; i++) {
148  // naively assign each qubit to some free node
149  placement.insert({to_place[i], difference[i]});
150  mapping_frontier->update_bimaps(
151  mapping_frontier->get_qubit_from_circuit_uid(to_place[i]),
152  difference[i]);
153  }
154 
155  mapping_frontier->update_linear_boundary_uids(placement);
156  mapping_frontier->circuit_.rename_units(placement);
157  }
158  }
159 
160  return circuit_modified;
161 }
