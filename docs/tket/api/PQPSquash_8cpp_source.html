38 PQPSquasher::PQPSquasher(OpType p, OpType q, bool smart_squash, bool reversed)
39  : p_(p),
40  q_(q),
41  smart_squash_(smart_squash),
42  reversed_(reversed),
43  rotation_chain() {
44  if (!(p == OpType::Rx || p == OpType::Ry || p == OpType::Rz) ||
45  !(q == OpType::Rx || q == OpType::Ry || q == OpType::Rz)) {
46  throw std::logic_error("Can only reduce chains of single qubit rotations");
47  }
48  if (p == q) {
49  throw std::logic_error(
50  "Requires two different bases to perform single qubit "
51  "rotations");
52  }
53 }

67 std::pair<Circuit, Gate_ptr> PQPSquasher::flush(
68  std::optional<Pauli> commutation_colour) const {
69  bool commute_through = false;
70  OpType p = p_, q = q_;
71 
72  if (smart_squash_ && commutation_colour.has_value()) {
73  // Using an arbitrary non-zero angle to obtain the commutation for p_/q_.
74  Gate P(p_, {0.123}, 1);
75  Gate Q(q_, {0.123}, 1);
76  if (P.commutes_with_basis(commutation_colour, 0)) {
77  commute_through = true;
78  } else if (Q.commutes_with_basis(commutation_colour, 0)) {
79  commute_through = true;
80  p = q_;
81  q = p_;
82  }
83  }
84 
85  // Construct list of merged rotations
86  std::list<Rotation> rots;
87  auto iter = rotation_chain.cbegin();
88  while (iter != rotation_chain.cend()) {
89  // Merge next q rotations
90  rots.push_back(merge_rotations(q, rotation_chain, iter));
91  // Merge next p rotations
92  rots.push_back(merge_rotations(p, rotation_chain, iter));
93  }
94 
95  // Perform any cancellations
96  std::list<Rotation>::iterator r = rots.begin();
97  while (r != rots.end()) {
98  if (r->is_id()) {
99  r = rots.erase(r);
100  if (r != rots.begin() && r != rots.end()) {
101  std::prev(r)->apply(*r);
102  r = rots.erase(r);
103  r--;
104  }
105  } else
106  r++;
107  }
108 
109  // Extract any P rotations from the beginning and end of the list
110  Expr p1 = 0, p2 = 0;
111  std::list<Rotation>::iterator i1 = rots.begin();
112  if (i1 != rots.end()) {
113  std::optional<Expr> a = i1->angle(p);
114  if (a) {
115  p1 = a.value();
116  rots.pop_front();
117  }
118  }
119  std::list<Rotation>::reverse_iterator i2 = rots.rbegin();
120  if (i2 != rots.rend()) {
121  std::optional<Expr> a = i2->angle(p);
122  if (a) {
123  p2 = a.value();
124  rots.pop_back();
125  }
126  }
127 
128  // Finish up:
129  Rotation R = {};
130  for (auto rot : rots) {
131  R.apply(rot);
132  }
133  std::tuple<Expr, Expr, Expr> angles = R.to_pqp(p, q);
134 
135  Expr angle_p1 = std::get<0>(angles) + p1;
136  Expr angle_q = std::get<1>(angles);
137  Expr angle_p2 = std::get<2>(angles) + p2;
138  fixup_angles(angle_p1, angle_q, angle_p2, reversed_);
139 
140  Circuit replacement(1);
141  Gate_ptr left_over_gate = nullptr;
142  if (!equiv_0(angle_p1, 4)) {
143  if (equiv_0(angle_q, 4) && equiv_0(angle_p2, 4) && commute_through) {
144  left_over_gate =
145  std::make_shared<Gate>(p, std::vector<Expr>{angle_p1}, 1);
146  } else {
147  replacement.add_op<unsigned>(p, angle_p1, {0});
148  }
149  }
150  if (!equiv_0(angle_q, 4)) {
151  replacement.add_op<unsigned>(q, angle_q, {0});
152  }
153  if (!equiv_0(angle_p2, 4)) {
154  if (commute_through) {
155  left_over_gate =
156  std::make_shared<Gate>(p, std::vector<Expr>{angle_p2}, 1);
157  } else {
158  replacement.add_op<unsigned>(p, angle_p2, {0});
159  }
160  }
161  redundancy_removal(replacement);
162  return {replacement, left_over_gate};
163 }

171 static Rotation merge_rotations(
172  OpType r, const std::vector<Gate_ptr> &chain,
173  std::vector<Gate_ptr>::const_iterator &iter) {
174  Expr total_angle(0);
175  while (iter != chain.end()) {
176  const Gate_ptr rot_op = *iter;
177  if (rot_op->get_type() != r) {
178  break;
179  }
180  total_angle += rot_op->get_params()[0];
181  iter++;
182  }
183  return Rotation(r, total_angle);
184 }

204 // To squash to TK1:
205 // - we first decompose to ZYZ. Doing this was found to reduce the size of
206 // symbolic expressions
207 // - we then redecompose to ZXZ, so that we can commute Rz or Rx rotation past
208 // multi-qubit gates (most usual multi-qb gates commute with X or Z)
209 // - Rz and Rx rotations can then be straight-forwardly combined into TK1s.
210 Transform squash_1qb_to_tk1() {
211  return Transforms::decompose_ZY() >>
212  squash_1qb_to_pqp(OpType::Ry, OpType::Rz, true) >>
213  Transforms::decompose_ZX() >>
214  squash_1qb_to_pqp(OpType::Rx, OpType::Rz, true) >>
215  Transforms::decompose_ZXZ_to_TK1();
216 }

218 static bool fixup_angles(
219  Expr &angle_p1, Expr &angle_q, Expr &angle_p2, bool reversed) {
220  bool success = false;
221  if (reversed) {
222  std::swap(angle_p1, angle_p2);
223  angle_p1 *= -1;
224  angle_q *= -1;
225  angle_p2 *= -1;
226  }
227  if (equiv_val(angle_q, 1., 2) && !equiv_0(angle_p2, 4)) {
228  // Prefer --P(p1-p2)--Q(...)--P(0)--
229  // Only occurs if angle_q is pi or 3pi and angle_p2 is non-zero
230  angle_p1 -= angle_p2;
231  angle_p2 = 0;
232  success = true;
233  } else if (equiv_val(angle_p2, 1., 4)) {
234  // Then prefer --P(p1+p2)--Q(-q)--P(0)--
235  // Only occurs if angle_p2 is pi
236  angle_p1 += 1;
237  angle_q *= -1;
238  angle_p2 = 0;
239  success = true;
240  } else if (equiv_val(angle_p2, 3., 4)) {
241  // Then prefer --P(p1+p2)--Q(-q)--P(0)--
242  // Only occurs if angle_p2 is 3pi
243  angle_p1 += 3;
244  angle_q *= -1;
245  angle_p2 = 0;
246  success = true;
247  } else if (equiv_val(angle_p1, 1., 4) && !equiv_0(angle_p2, 4)) {
248  // Then prefer --P(0)--Q(-q)--P(p1+p2)--
249  // Only occurs if angle_p1 is pi and angle_p2 is non-zero
250  angle_q *= -1;
251  angle_p2 += 1;
252  angle_p1 = 0;
253  success = true;
254  } else if (equiv_val(angle_p1, 3., 4) && !equiv_0(angle_p2, 4)) {
255  // Then prefer --P(0)--Q(-q)--P(p1+p2)--
256  // Only occurs if angle_p1 is 3pi and angle_p2 is non-zero
257  angle_q *= -1;
258  angle_p2 += 3;
259  angle_p1 = 0;
260  success = true;
261  }
262  if (reversed) {
263  std::swap(angle_p1, angle_p2);
264  angle_p1 *= -1;
265  angle_q *= -1;
266  angle_p2 *= -1;
267  }
268  return success;
269 }

271 static bool remove_redundancy(
272  Circuit &circ, const Vertex &vert, VertexList &bin,
273  std::set<IVertex> &new_affected_verts, IndexMap &im) {
274  const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
275  const OpDesc desc = op->get_desc();
276  if (!desc.is_gate()) return false;
277  if (circ.n_out_edges(vert) == 0 || circ.n_in_edges(vert) == 0) {
278  return false; // either a boundary vert or we have already removed it
279  }
280 
281  auto remove_single_vertex = [&bin, &circ, &new_affected_verts,
282  &im](const Vertex &v_remove) {
283  bin.push_back(v_remove);
284  for (const Vertex &l : circ.get_predecessors(v_remove)) {
285  new_affected_verts.insert({im.at(l), l});
286  }
287  circ.remove_vertex(
288  v_remove, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
289  };
290  // remove 0 angle rotations from circuit
291  std::optional<double> a = op->is_identity();
292  if (a) {
293  remove_single_vertex(vert);
294  circ.add_phase(a.value());
295  return true;
296  } else if (desc.type() == OpType::noop) {
297  // remove "noop" gates from circuit
298  remove_single_vertex(vert);
299  return true;
300  }
301  VertexVec kids = circ.get_successors(vert);
302 
303  // if op is immediately followed by a z basis measurement on all qubits,
304  // remove
305  if (circ.n_out_edges_of_type(vert, EdgeType::Classical) == 0) {
306  bool z_followed_by_measures = true;
307  for (port_t port = 0; port < kids.size() && z_followed_by_measures;
308  port++) {
309  if (circ.get_OpType_from_Vertex(kids[port]) == OpType::Measure) {
310  z_followed_by_measures &=
311  circ.commutes_with_basis(vert, Pauli::Z, PortType::Source, port);
312  } else {
313  z_followed_by_measures = false;
314  }
315  }
316  if (z_followed_by_measures) {
317  remove_single_vertex(vert);
318  return true;
319  }
320  }
321 
322  // check that both the vertex and its successor have each other and only each
323  // other
324  if ((kids.size() == 1) && (circ.get_predecessors(kids[0]).size() == 1)) {
325  // check that the ports match up between vertices
326  Vertex b = kids[0];
327  EdgeVec ins = circ.get_in_edges(b);
328  for (const Edge &in : ins) {
329  if (circ.get_source_port(in) != circ.get_target_port(in)) return false;
330  }
331 
332  // check that the classical edges match up correctly
333  if (circ.n_in_edges_of_type(vert, EdgeType::Boolean) != 0) return false;
334 
335  const Op_ptr b_op = circ.get_Op_ptr_from_Vertex(b);
336  const OpDesc b_desc = b_op->get_desc();
337 
338  if (!b_desc.is_oneway()) {
339  // if A = B.dagger(), AB = I
340  // This method cannot detect matches between rotation gates.
341  // Rotation gates are covered by the rotation gate combiner, everything
342  // else in this.
343  if (*b_op->dagger() == *op) {
344  bin.push_back(vert);
345  bin.push_back(b);
346  VertexVec last_verts = circ.get_predecessors(vert);
347  for (const Vertex &l : last_verts) {
348  new_affected_verts.insert({im.at(l), l});
349  }
350  VertexList to_detach{vert, b};
351  // detached from circuit but not removed from graph
352  circ.remove_vertices(
353  to_detach, Circuit::GraphRewiring::Yes,
354  Circuit::VertexDeletion::No);
355  return true;
356  } else if (desc.is_rotation()) {
357  // combine two rotation gates together, then if the combined
358  // operation is the identity up to phase, remove from circuit
359  if (b_desc.type() == desc.type()) {
360  Expr expr1 = op->get_params()[0];
361  Expr expr2 = b_op->get_params()[0];
362  VertexVec last_verts = circ.get_predecessors(vert);
363  for (const Vertex &l : last_verts) {
364  new_affected_verts.insert({im.at(l), l});
365  }
366  circ.remove_vertex(
367  b, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
368  bin.push_back(b);
369  std::vector<Expr> params_new = {expr1 + expr2};
370  Op_ptr op_new = get_op_ptr(desc.type(), params_new, ins.size());
371  std::optional<double> a = op_new->is_identity();
372  if (a) {
373  bin.push_back(vert);
374  circ.remove_vertex(
375  vert, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
376  circ.add_phase(a.value());
377  } else {
378  new_affected_verts.insert({im[vert], vert});
379  circ.dag[vert].op = op_new;
380  }
381  return true;
382  }
383  }
384  }
385  }
386  return false;
387 }

389 static bool redundancy_removal(Circuit &circ) {
390  bool success = false;
391  bool found_redundancy = true;
392  IndexMap im = circ.index_map();
393  std::set<IVertex> old_affected_verts;
394  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
395  old_affected_verts.insert({im.at(v), v});
396  }
397  VertexList bin;
398  while (found_redundancy) {
399  std::set<IVertex> new_affected_verts;
400  for (const IVertex &p : old_affected_verts) {
401  remove_redundancy(circ, p.second, bin, new_affected_verts, im);
402  }
403  found_redundancy = new_affected_verts.size() != 0;
404  success |= found_redundancy;
405  old_affected_verts = new_affected_verts;
406  }
407  circ.remove_vertices(
408  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
409  return success;
410 }
