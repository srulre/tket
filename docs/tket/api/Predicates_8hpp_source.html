47 // abstract interface class
48 class Predicate {
49  public:
50  virtual bool verify(const Circuit& circ) const = 0;
51 
52  // implication currently only works between predicates of the same subclass
53  virtual bool implies(const Predicate& other) const = 0;
54  virtual PredicatePtr meet(const Predicate& other) const = 0;
55  virtual std::string to_string() const = 0;
56  virtual ~Predicate(){}; // satisfy compiler
57 };
