19 // Add a swicth with the on-state controlled by the on_value.
20 // qtype indicates the quantum type of the switch.
21 std::pair<ZXVertPort, ZXVertPort> add_switch(
22  ZXDiagram& zxd, const bool& on_value, const QuantumType& qtype) {
23  zxd.multiply_scalar(std::sqrt(2.));
24  ZXVert triangle = zxd.add_vertex(ZXType::Triangle, qtype);
25  ZXVert x = zxd.add_vertex(ZXType::XSpider, 0, qtype);
26  zxd.add_wire(triangle, x, ZXWireType::Basic, qtype, 1);
27  if (on_value) {
28  // Turn on the switch if the input is 1
29  ZXVert negate = zxd.add_vertex(ZXType::XSpider, 1, qtype);
30  zxd.add_wire(triangle, negate, ZXWireType::Basic, qtype, 0);
31  return {{negate, std::nullopt}, {x, std::nullopt}};
32  }
33  return {{triangle, 0}, {x, std::nullopt}};
34 }

51  ZXVert out = zxd.add_vertex(ZXType::ZSpider, 0, qtype);

59  zxd.add_wire(out, switch_s0.second.first, ZXWireType::Basic, qtype);
60  zxd.add_wire(in, switch_n0.second.first, ZXWireType::Basic, qtype);
61  zxd.add_wire(left, switch_n0.second.first, ZXWireType::Basic, qtype);
62  zxd.add_wire(right, discard_ctr, ZXWireType::Basic, qtype);
63  zxd.add_wire(discard_ctr, switch_n1.second.first, ZXWireType::Basic, qtype);
64  zxd.add_wire(out, switch_n1.second.first, ZXWireType::Basic, qtype);
65  zxd.add_wire(discard_ctr, switch_s1.second.first, ZXWireType::Basic, qtype);
66  zxd.add_wire(discard, switch_s1.second.first, ZXWireType::Basic, qtype);
67  return {
68  {{in, std::nullopt}, {out, std::nullopt}},
69  {switch_s0.first, switch_s1.first, switch_n0.first, switch_n1.first}};
70 }

262  {{{vert, 0}, PortType::In}, {zx_vert, std::nullopt}});

288  // Find the nested quantum op and its overall conditions

506  vert_lookup.at(TypedVertPort(VertPort(v_s, p_s), PortType::Out));
507  ZXVertPort zx_vp_t =
508  vert_lookup.at(TypedVertPort(VertPort(v_t, p_t), PortType::In));
509  if (circ.get_edgetype(edge) == EdgeType::Quantum) {
510  zxd.add_wire(
511  zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic, QuantumType::Quantum,
512  zx_vp_s.second, zx_vp_t.second);
513  } else {
514  auto bool_it = boolean_outport_lookup.find(VertPort(v_s, p_s));
515  if (bool_it == boolean_outport_lookup.end()) {
516  zxd.add_wire(
517  zx_vp_s.first, zx_vp_t.first, ZXWireType::Basic,
518  QuantumType::Classical, zx_vp_s.second, zx_vp_t.second);
519  } else {
520  // If the source port is boolean, then connect the copy spider to the
521  // source vertex. All out-edges originated from the source port should
522  // be connected to the copy spider.
523  if (zxd.degree(bool_it->second.first) == 0) {
524  zxd.add_wire(
525  zx_vp_s.first, bool_it->second.first, ZXWireType::Basic,
526  QuantumType::Classical, zx_vp_s.second, bool_it->second.second);
527  }
528  zxd.add_wire(
529  bool_it->second.first, zx_vp_t.first, ZXWireType::Basic,
530  QuantumType::Classical, bool_it->second.second, zx_vp_t.second);
531  }
532  }
533  }
534  return bmap;
535 }

537 std::pair<ZXDiagram, BoundaryVertMap> circuit_to_zx(const Circuit& circ) {
538  ZXDiagram zxd;
539  BoundaryVertMap bmap = circuit_to_zx_recursive(circ, zxd, true);
540  // Remove internal boundary vertices produced by the recursion
541  ZXVertVec true_boundary = zxd.get_boundary();
542  ZXVertIterator vi, vi_end, next;
543  tie(vi, vi_end) = boost::vertices(*zxd.get_graph());
544  for (next = vi; vi != vi_end; vi = next) {
545  ++next;
546  if ((zxd.get_zxtype(*vi) == ZXType::Input ||
547  zxd.get_zxtype(*vi) == ZXType::Output) &&
548  std::find(true_boundary.begin(), true_boundary.end(), *vi) ==
549  true_boundary.end()) {
550  WireVec adj_wires = zxd.adj_wires(*vi);
551  TKET_ASSERT(adj_wires.size() == 2);
552  TKET_ASSERT(zxd.get_qtype(adj_wires[0]) == zxd.get_qtype(adj_wires[1]));
553  TKET_ASSERT(
554  zxd.get_wire_type(adj_wires[0]) == zxd.get_wire_type(adj_wires[1]));
555  ZXVertVec neighbours = zxd.neighbours(*vi);
556  zxd.add_wire(
557  neighbours[0], neighbours[1], zxd.get_wire_type(adj_wires[0]),
558  zxd.get_qtype(adj_wires[0]));
559  zxd.remove_vertex(*vi);
560  }
561  }
562  return {std::move(zxd), std::move(bmap)};
563 }
