27 std::set<Qubit> interacting_qbs(const Circuit& circ) {
28  std::set<Qubit> qbs;
29  for (const Qubit& qb : circ.all_qubits()) {
30  Edge e = circ.get_nth_out_edge(circ.get_in(qb), 0);
31  Vertex terminal = circ.target(circ.skip_irrelevant_edges(e));
32  if (!circ.detect_final_Op(terminal)) {
33  qbs.insert(qb);
34  }
35  }
36 
37  return qbs;
38 }

40 PlacementFrontier::PlacementFrontier(const Circuit& _circ) : circ(_circ) {
41  VertexVec input_slice;
42  quantum_in_edges = std::make_shared<unit_frontier_t>();
43  boolean_in_edges = std::make_shared<b_frontier_t>();
44 
45  for (const Qubit& qb : circ.all_qubits()) {
46  Vertex input = circ.get_in(qb);
47  input_slice.push_back(input);
48  Edge candidate = circ.get_nth_out_edge(input, 0);
49  quantum_in_edges->insert({qb, circ.skip_irrelevant_edges(candidate)});
50  }
51  for (const Bit& bit : circ.all_bits()) {
52  Vertex input = circ.get_in(bit);
53  EdgeVec candidates = circ.get_nth_b_out_bundle(input, 0);
54  boolean_in_edges->insert({bit, candidates});
55  }
56 
57  CutFrontier next_cut = circ.next_cut(quantum_in_edges, boolean_in_edges);
58  slice = next_cut.slice;
59  quantum_out_edges = next_cut.u_frontier;
60 }

62 void PlacementFrontier::next_slicefrontier() {
63  quantum_in_edges = std::make_shared<unit_frontier_t>();
64  boolean_in_edges = std::make_shared<b_frontier_t>();
65  for (const std::pair<UnitID, Edge>& pair : quantum_out_edges->get<TagKey>()) {
66  Edge new_e = circ.skip_irrelevant_edges(pair.second);
67  quantum_in_edges->insert({pair.first, new_e});
68  Vertex targ = circ.target(new_e);
69  EdgeVec targ_classical_ins =
70  circ.get_in_edges_of_type(targ, EdgeType::Boolean);
71  boolean_in_edges->insert(
72  {Bit("frontier_bit", pair.first.index()), targ_classical_ins});

75  CutFrontier next_cut = circ.next_cut(quantum_in_edges, boolean_in_edges);
76  slice = next_cut.slice;
77  quantum_out_edges = next_cut.u_frontier;
78 }

80 QubitGraph monomorph_interaction_graph(
81  const Circuit& circ, const unsigned max_edges, unsigned depth_limit) {
82  std::set<Qubit> qubits_considered = interacting_qbs(circ);
83 
84  QubitGraph q_graph(circ.all_qubits());
85 
86  PlacementFrontier current_sf(circ);
87  unsigned count_edges = 0;
88  for (unsigned slice = 0;
89  slice < depth_limit && count_edges < max_edges &&
90  !current_sf.slice->empty() && qubits_considered.size() > 1;
91  slice++) {
92  for (const Vertex& vert : *current_sf.slice) {
93  EdgeVec q_out_edges = circ.get_out_edges_of_type(vert, EdgeType::Quantum);
94  Qubit qb1;
95  Qubit qb2;
96  for (const std::pair<UnitID, Edge>& pair :
97  current_sf.quantum_out_edges->get<TagKey>()) {
98  if (pair.second == q_out_edges[0])
99  qb1 = Qubit(pair.first);
100  else if (pair.second == q_out_edges[1])
101  qb2 = Qubit(pair.first);
102  }
103  if (!q_graph.edge_exists(qb1, qb2) && !q_graph.edge_exists(qb2, qb1)) {
104  q_graph.add_connection(qb1, qb2, slice + 1);
105  count_edges++;
106  }
107  }
108 
109  current_sf.next_slicefrontier();
110  }
111  q_graph.remove_stray_nodes();
112  return q_graph;
113 }

115 QubitGraph generate_interaction_graph(
116  const Circuit& circ, unsigned depth_limit) {
117  std::set<Qubit> qubits_considered = interacting_qbs(circ);
118  QubitGraph q_graph(circ.all_qubits());
119  PlacementFrontier current_sf(circ);
120 
121  for (unsigned slice = 0; slice < depth_limit && !current_sf.slice->empty() &&
122  qubits_considered.size() > 1;
123  slice++) {
124  for (const Vertex& vert : *current_sf.slice) {
125  EdgeVec q_out_edges = circ.get_out_edges_of_type(vert, EdgeType::Quantum);
126  Qubit qb1;
127  Qubit qb2;
128  for (const std::pair<UnitID, Edge>& pair :
129  current_sf.quantum_out_edges->get<TagKey>()) {
130  if (pair.second == q_out_edges[0])
131  qb1 = Qubit(pair.first);
132  else if (pair.second == q_out_edges[1])
133  qb2 = Qubit(pair.first);
134  }
135  const bool qb1_considered =
136  (qubits_considered.find(qb1) != qubits_considered.end());
137  const bool qb2_considered =
138  (qubits_considered.find(qb2) != qubits_considered.end());
139  if ((qb2 != qb1) && (qb1_considered || qb2_considered)) {
140  if (!qb1_considered) {
141  qubits_considered.erase(qb2);
142  } else if (!qb2_considered) {
143  qubits_considered.erase(qb1);
144  } else if (!q_graph.edge_exists(qb1, qb2)) {
145  const unsigned out1 = q_graph.get_degree(qb1);
146  const unsigned out2 = q_graph.get_degree(qb2);
147  q_graph.add_connection(qb1, qb2, slice + 1);
148  if (out1 == 1) {
149  qubits_considered.erase(qb1);
150  }
151  if (out2 == 1) {
152  qubits_considered.erase(qb2);
153  }
154  } else {
155  qubits_considered.erase(qb1);
156  qubits_considered.erase(qb2);
157  }
158  }
159  }
160  current_sf.next_slicefrontier();
161  }
162  q_graph.remove_stray_nodes();
163 
164  return q_graph;
165 }

167 QubitLineList qubit_lines(const Circuit& circ) {
168  const QubitGraph q_graph = generate_interaction_graph(circ);
169  std::set<Qubit> all_qb;
170  for (const Qubit& qb : circ.all_qubits()) {
171  all_qb.insert(qb);
172  }
173  QubitLineList found_lines;
174  unsigned found_line_size = 0;
175  QubitGraph::UndirectedConnGraph graph = q_graph.get_undirected_connectivity();
176  do {
177  auto u_line = graphs::longest_simple_path(graph);
178  QubitLine found;
179  std::transform(
180  u_line.begin(), u_line.end(), std::back_inserter(found),
181  [&graph](auto v) -> Qubit { return graph[v]; });
182  found_line_size = found.size();
183  if (found_line_size > 1) {
184  found_lines.push_back(found);
185  for (const auto& vertex : u_line) {
186  boost::clear_vertex(vertex, graph);
187  }
188  for (Qubit q : found) {
189  all_qb.erase(q);
190  }
191  }
192  } while (found_line_size > 1);
193 
194  for (const Qubit& qb : circ.all_qubits()) {
195  if (all_qb.find(qb) != all_qb.end()) found_lines.push_back({qb});
196  }
197  // print_qubitlines(found_lines);
198  return found_lines;
199 }

201 // remove a given number of nodes from the architecture, return a set of usable
202 // nodes (the remainder)
203 node_set_t best_nodes(Architecture& arc, unsigned n_remove) {
204  node_set_t all_nodes = arc.nodes();
205  node_set_t bad_nodes;
206  // if there are nodes 'removed' already count them as bad nodes
207  for (Node n : all_nodes) {
208  if (arc.get_degree(n) == 0) {
209  bad_nodes.insert(n);
210  n_remove--;
211  }
212  }
213  node_set_t removed_nodes = arc.remove_worst_nodes(n_remove);
214  bad_nodes.insert(removed_nodes.begin(), removed_nodes.end());
215  node_set_t good_nodes; // keep track of nodes of architecture actually used
216  std::set_difference(
217  all_nodes.begin(), all_nodes.end(), bad_nodes.begin(), bad_nodes.end(),
218  std::inserter(good_nodes, good_nodes.begin()));
219  return good_nodes;
220 }

222 // map qubit lines to node lines, erasing qubit lines as it goes
223 qubit_mapping_t map_lines(
224  QubitLineList& qb_lines, const std::vector<node_vector_t>& node_lines) {
225  qubit_mapping_t outmap;
226 
227  // go through all node lines
228  for (unsigned i = 0; i < node_lines.size(); i++) {
229  node_vector_t node_lst = node_lines[i];
230  for (const Node& node : node_lst) {
231  outmap.insert({*qb_lines[i].begin(), node});
232  qb_lines[i].erase(qb_lines[i].begin());
233  }
234  }
235  return outmap;
236 }

238 // trivially place qubit lines on to available nodes, return map
239 qubit_mapping_t place_qubit_lines(
240  const QubitLineList& qb_lines, node_set_t available_nodes) {
241  qubit_mapping_t outmap;
242 
243  node_set_t::iterator node_it = available_nodes.begin();
244  for (const QubitLine& line : qb_lines) {
245  for (const Qubit& qb : line) {
246  if (node_it == available_nodes.end()) {
247  throw ArchitectureInvalidity("Not enough nodes to place all qubits.");
248  }
249  outmap.insert({qb, *node_it});
250  ++node_it;
251  }
252  }
253  return outmap;
254 }

256 // Determine intial qubit placement by requesting lines of architecture to place
257 // lines of qubits on
258 qubit_mapping_t lines_on_arc(
259  Architecture arc, QubitLineList qb_lines, unsigned n_qubits) {
260  unsigned difference = arc.n_nodes() - n_qubits;
261  // sort from longest to shortest
262  std::sort(qb_lines.begin(), qb_lines.end(), [](QubitLine x, QubitLine y) {
263  return (x.size() > y.size());
264  });
265 
266  // get rid of one qubit lines
267  while (!qb_lines.empty() && qb_lines.back().size() < 2) {
268  difference++;
269  qb_lines.pop_back();
270  }
271 
272  // remove poorly connected nodes, up to the number not used by mapping
273  node_set_t unused_nodes = best_nodes(arc, difference);
274 
275  // find lengths required
276  std::vector<unsigned> lengths;
277  for (const QubitLine& line : qb_lines) {
278  lengths.push_back(line.size());
279  }
280 
281  // attempt to find lines of required length on architecture
282  std::vector<node_vector_t> node_lines = arc.get_lines(lengths);
283 
284  // map qubit lines to node lines to some extent
285  qubit_mapping_t outmap = map_lines(qb_lines, node_lines);
286  for (qubit_mapping_t::iterator it = outmap.begin(); it != outmap.end();
287  ++it) {
288  unused_nodes.erase(it->second);
289  }
290  // map remain remaining qubit lines trivially
291  qubit_mapping_t remainder_map = place_qubit_lines(qb_lines, unused_nodes);
292  outmap.insert(remainder_map.begin(), remainder_map.end());
293 
294  return outmap;
295 }

309 // calculate a cost value for map being considered
310 double Monomorpher::map_cost(const qubit_bimap_t& n_map) {
311  double cost = 0.0;
312  const int approx_depth = circ.n_gates() / circ.n_qubits() + 1;
313  // constants for scaling single qubit error
314  constexpr double c1 = 0.5;
315  constexpr double d1 = 1 - 1 / c1;
316  for (auto [qb, node] : n_map) {
317  // add fidelities of edges from node, weighted by if edge is used by
318  // interaction graph
319  node_set_t neighs = arc.get_neighbour_nodes(node);
320  double edge_sum = 1.0;
321  for (Node nei : neighs) {
322  double fwd_edge_weighting = 1.0, bck_edge_weighting = 1.0;
323  // check if neighbour node is mapped
324  r_const_iterator_t nei_qb_it = n_map.right.find(nei);
325  if (nei_qb_it == n_map.right.end()) continue;
326  // if ( nei_qb_it != n_map.end()){
327  unsigned edge_val = q_graph.get_connection_weight(qb, nei_qb_it->second);
328  // check if either directed interaction exists
329  auto place_interactions_boost = [&](unsigned edge_v) {
330  return config.depth_limit - edge_v + 1;
331  };
332  // if edge is used by interaction in mapping, weight edge higher
333  if (edge_val) {
334  fwd_edge_weighting += place_interactions_boost(edge_val);
335  } else {
336  edge_val = q_graph.get_connection_weight(nei_qb_it->second, qb);
337  if (edge_val) {
338  bck_edge_weighting += place_interactions_boost(edge_val);
339  }
340  }
341  // }
342  gate_error_t fwd_error = characterisation.get_error({node, nei});
343  gate_error_t bck_error = characterisation.get_error({nei, node});
344  edge_sum += fwd_edge_weighting * (1.0 - fwd_error);
345  edge_sum += bck_edge_weighting * (1.0 - bck_error);
346  }
347 
348  // bigger edge sum -> smaller cost
349  cost += 1.0 / (edge_sum);
350 
351  // add error rate of node
352  gate_error_t single_error = characterisation.get_error(node);
353  cost += d1 + 1.0 / ((1.0 - single_error) + c1);
354  readout_error_t readout_error = characterisation.get_readout_error(node);
355  cost += (d1 + 1.0 / ((1.0 - readout_error) + c1)) / (approx_depth * 20);
356  // TODO add readout weighting to PlacementConfig?
357  }
358 
359  return cost;
360 }

362 std::vector<MapCost> Monomorpher::place(unsigned max_return) {
363  if (max_return < 1)
364  throw PlacementError("Max return maps for place must be at least 1.");
365  const unsigned interacting_nodes = q_graph.n_connected();
366  if ((circ.n_gates() / circ.n_qubits()) >= config.arc_contraction_ratio &&
367  circ.n_qubits() > 3) {
368  best_nodes(arc, arc.n_nodes() - interacting_nodes);
369  }
370 
371  std::vector<qubit_bimap_t> potential_maps = monomorphism_edge_break(
372  arc, q_graph, config.vf2_max_matches, config.timeout);
373  std::vector<MapCost> map_costs;
374  for (unsigned i = 0; i < potential_maps.size(); i++) {
375  qubit_bimap_t& chosen = potential_maps[i];
376  double cost = map_cost(chosen);
377  qubit_mapping_t converted_map;
378  for (auto [qb, node] : chosen.left) {
379  converted_map[qb] = node;
380  }
381  map_costs.push_back({converted_map, cost});
382  }
383 
384  max_return = std::min(static_cast<unsigned>(map_costs.size()), max_return);
385  std::nth_element(
386  map_costs.begin(), map_costs.begin() + max_return - 1, map_costs.end());
387  const MapCost nth_map = *(map_costs.begin() + max_return - 1);
388  for (std::vector<MapCost>::iterator it = map_costs.begin();
389  it != map_costs.end();) {
390  if (*it > nth_map) {
391  it = map_costs.erase(it);
392  } else {
393  ++it;
394  }
395  }
396 
397  if (max_return < map_costs.size()) map_costs.resize(max_return);
398  return map_costs;
399 }
