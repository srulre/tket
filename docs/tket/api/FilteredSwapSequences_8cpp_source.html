29 /*
30 NOTE: the problem is: given a bitset, i.e. an unsigned int representing a set,
31 design a map-type data structure whose keys are unsigned integers representing
32 bitsets, and values are a collection of entries using that bitset (i.e., only
33 using swaps whose index in a global vector of allowed swaps has a "one" in the
34 appropriate position in the binary expansion of the bitset).
35 
36 We must be able to look up all entries whose key is a SUBSET of the given set.
37 (And then, search further through those values).
38 
39 We tried various things, e.g. sorting by key, using the fact that
40 
41 (X is a subset of Y) ==> (x <= y)
42 
43 where X,Y are subsets and x,y are the integers representing them; thus you can
44 do a kind of binary search.
45 
46 If you want the SMALLEST value for a given key, you can sort them also and do a
47 kind of double binary search. (Another crucial point: when searching between two
48 key ranges in a sorted VECTOR of keys, you can determine how many keys exist in
49 the range in O(log N) time, rather than O(N) time for a map).
50 
51 These fancy algorithms are all asymptotically much more efficient than the
52 obvious O(N) lookup, which just goes through EVERY key and checks if it's a
53 subset or not, then goes through every element.
54 
55 HOWEVER, experiments showed that the fancy algorithms are actually quite a bit
56 slower than the obvious algorithm for the table size we care about.
57 
58 */

65 /*
66 If the entries are distributed "randomly" and fairly uniformly amongst the
67 bitset keys, i.e. given a bitset, look up all keys which are a subset of that,
68 then asymptotically using many bits in the keys is good.
69 
70 For our table sizes, experiments suggested that it's worth having 1 bit in each
71 bitset key (2 min for 1 bit vs. 2 min 20 sec for no bits in one test), rather
72 then no keys at all, BUT not worth more than 1 bit in each key.
73 
74 e.g., for 15 bits in each bitset, each of the 15 keys being one of the bits
75 (we have no empty keys - pointless trying to look up swap sequences if the graph
76 has no edges!), assume that an average lookup query contains 5 bits. Then 10/15
77 = 2/3 of the keys are disjoint from it, and so most of the keys immediately can
78 be ruled out.
79 
80 However, it's a balancing act: if you have too many keys, then the lists for
81 each key become so short then you're effectively almost doing a linear search
82 through all entries.
83 */

85 void FilteredSwapSequences::initialise(
86  std::vector<SwapConversion::SwapHash> codes) {
87  // Can only initialise once.
88  TKET_ASSERT(m_internal_data.empty());
89  std::sort(codes.begin(), codes.end());
90  TKET_ASSERT(!codes.empty());
91  TKET_ASSERT(codes[0] != 0);
92  TrimmedSingleSequenceData datum;
93 
94  for (size_t ii = 0; ii < codes.size(); ++ii) {
95  if (ii != 0 && codes[ii] == codes[ii - 1]) {
96  // Filter out duplicate entries.
97  continue;
98  }
99  datum.swaps_code = codes[ii];
100  datum.edges_bitset = SwapConversion::get_edges_bitset(datum.swaps_code);
101  push_back(datum);
102  }
103 }

105 void FilteredSwapSequences::push_back(TrimmedSingleSequenceData datum) {
106  auto bitset_copy = datum.edges_bitset;
107  TKET_ASSERT(bitset_copy != 0);
108  SwapConversion::EdgesBitset bit_to_use = 0;
109 
110  // We want to add to the smallest list, to keep the data balanced.
111  // Tests showed that this works well; the entries are distributed
112  // very close to uniformly amongst the 15 possible keys.
113  //
114  // This is maybe surprising, because you'd expect
115  // more bias: you'd expect, due to the relabelling scheme, the table to have
116  // swaps like (0,1), (0,2) much more frequently than higher-numbered
117  // vertices like (4,5). This may or may not be the case, but whatever
118  // the truth, there are still enough bits available overall to break
119  // the entries up well enough).
120  size_t list_size_to_use = std::numeric_limits<size_t>::max();
121 
122  while (bitset_copy != 0) {
123  const auto new_bit = get_rightmost_bit(bitset_copy);
124  // If the key does not exist, the newly created empty list will
125  // immediately be filled; so no key is wasted. (They're not wasted anyway,
126  // the table entries are very close to uniformly distributed
127  // amongst all 15 keys).
128  const auto list_size = m_internal_data[new_bit].size();
129 
130  if (list_size < list_size_to_use) {
131  list_size_to_use = list_size;
132  bit_to_use = new_bit;
133  if (list_size == 0) {
134  break;
135  }
136  }
137  }
138  TKET_ASSERT(bit_to_use != 0);
139  m_internal_data[bit_to_use].push_back(datum);
140 }

142 FilteredSwapSequences::SingleSequenceData
143 FilteredSwapSequences::get_lookup_result(
144  SwapConversion::EdgesBitset edges_bitset, unsigned max_num_swaps) const {
145  // NOTE: this algorithm is quite crude, BUT it's so simple that
146  // apparently clever algorithms, although asymptotically more efficient,
147  // appear to be slower.
148  // The clever algorithms seem only worth doing if the table becomes
149  // much larger, >> 100 codes for each bit at least.
150 
151  max_num_swaps = std::min(max_num_swaps, 16u);
152 
153  // Value 0xFFF...F will never occur,
154  // because this would be 16 consecutive equal swaps...!
155  const auto impossible_max_code =
156  std::numeric_limits<SwapConversion::SwapHash>::max();
157 
158  // Stop as soon as the swaps code gets too big.
159  SwapConversion::SwapHash max_code;
160  if (max_num_swaps == 16) {
161  max_code = impossible_max_code;
162  } else {
163  max_code = 1;
164  max_code <<= (4 * max_num_swaps);
165  --max_code;
166  }
167  TrimmedSingleSequenceData best_datum;
168  best_datum.swaps_code = impossible_max_code;
169 
170  for (const auto& entry : m_internal_data) {
171  if (entry.first > edges_bitset) {
172  // The swaps used by a sequence must be a SUBSET of the allowable edges.
173  // Therefore, the swaps bitset must be <= the edges bitset.
174  // Of course, it's a MAP, so the swaps bitsets are already in increasing
175  // order.
176  break;
177  }
178  if ((entry.first & edges_bitset) != entry.first) {
179  // Every swap sequence in this entry contains ALL of the given edges
180  // in the bitset key (as well as others), and thus it MUST be a subset
181  // of the given edges_bitset, otherwise the entire entry
182  // can be skipped.
183  continue;
184  }
185  const auto& list = entry.second;
186  for (const auto& single_entry : list) {
187  if (single_entry.swaps_code > max_code ||
188  single_entry.swaps_code >= best_datum.swaps_code) {
189  // Because they're sorted by code value,
190  // all subsequent entries will be too big also.
191  break;
192  }
193  if ((single_entry.edges_bitset & edges_bitset) !=
194  single_entry.edges_bitset) {
195  // The EXACT set of edges used must be a subset of edges_bitset,
196  // otherwise it's unsuitable - it uses a swap not allowed.
197  continue;
198  }
199  best_datum = single_entry;
200  }
201  }
202 
203  SingleSequenceData result;
204  if (best_datum.swaps_code < impossible_max_code) {
205  // We actually got a result.
206  result.edges_bitset = best_datum.edges_bitset;
207  result.swaps_code = best_datum.swaps_code;
208  result.number_of_swaps =
209  SwapConversion::get_number_of_swaps(result.swaps_code);
210  }
211  return result;
212 }

222 // Convert the raw SwapSequenceTable object into
223 // FilteredSwapSequences-compatible data. The key is the permutation hash; the
224 // value is the lookup object which can find solutions to given problems.
225 static std::map<unsigned, FilteredSwapSequences>
226 construct_and_return_full_table() {
227  std::map<unsigned, FilteredSwapSequences> result;
228  const auto raw_table = SwapSequenceTable::get_table();
229  for (const auto& entry : raw_table) {
230  // The simplest nontrivial permutation arises from a single swap (a,b),
231  // which under the canonical relabelling is converted to (01),
232  // which has hash 2.
233  TKET_ASSERT(entry.first >= 2);
234  // The largest possible hash comes from (01)(23)(45).
235  TKET_ASSERT(entry.first <= 222);
236  result[entry.first].initialise(entry.second);
237  }
238  return result;
239 }

241 static const std::map<unsigned, FilteredSwapSequences>& get_full_table() {
242  static const auto full_table(construct_and_return_full_table());
243  return full_table;

246 FilteredSwapSequences::SingleSequenceData::SingleSequenceData(
247  unsigned permutation_hash, SwapConversion::EdgesBitset edges_bitset,
248  unsigned max_number_of_swaps)
249  : SingleSequenceData() {
250  if (permutation_hash == 0) {
251  // The identity mapping, always possible.
252  number_of_swaps = 0;
253  return;
254  }
255  if (edges_bitset == 0) {
256  // No swaps at all! This CAN happen...it just means that
257  // we haven't seen enough vertices to connect up the given ones;
258  // all solutions involve swaps using other vertices not yet seen
259  // (i.e., not in this subgraph).
260  // But it's not the identity, therefore it's impossible.
261  return;
262  }
263 
264  const auto& table = get_full_table();
265  const auto citer = table.find(permutation_hash);
266  if (citer == table.cend()) {
267  // No result in the table.
268  return;
269  }
270  *this = citer->second.get_lookup_result(edges_bitset, max_number_of_swaps);
271 }
