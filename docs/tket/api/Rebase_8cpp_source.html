46 static bool standard_rebase(
47  Circuit& circ, const OpTypeSet& allowed_gates,
48  const Circuit& cx_replacement,
49  const std::function<Circuit(const Expr&, const Expr&, const Expr&)>&
50  tk1_replacement) {
51  bool success = false;
52  VertexList bin;
53  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
54  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
55  unsigned n_qubits = circ.n_in_edges_of_type(v, EdgeType::Quantum);
56  if (n_qubits <= 1) continue;
57  bool conditional = op->get_type() == OpType::Conditional;
58  if (conditional) {
59  const Conditional& cond = static_cast<const Conditional&>(*op);
60  op = cond.get_op();
61  }
62  OpType type = op->get_type();
63  if (allowed_gates.find(type) != allowed_gates.end() || type == OpType::CX ||
64  type == OpType::Barrier)
65  continue;
66  // need to convert
67  Circuit replacement = CX_circ_from_multiq(op);
68  if (conditional) {
69  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
70  } else {
71  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
72  }
73  bin.push_back(v);
74  success = true;
75  }
76  if (allowed_gates.find(OpType::CX) == allowed_gates.end()) {
77  const Op_ptr cx_op = get_op_ptr(OpType::CX);
78  success = circ.substitute_all(cx_replacement, cx_op) | success;
79  }
80  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
81  if (circ.n_in_edges_of_type(v, EdgeType::Quantum) != 1) continue;
82  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
83  bool conditional = op->get_type() == OpType::Conditional;
84  if (conditional) {
85  const Conditional& cond = static_cast<const Conditional&>(*op);
86  op = cond.get_op();
87  }
88  OpType type = op->get_type();
89  if (!is_gate_type(type) || is_projective_type(type) ||
90  allowed_gates.find(type) != allowed_gates.end())
91  continue;
92  // need to convert
93  std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
94  Circuit replacement =
95  tk1_replacement(tk1_angles[0], tk1_angles[1], tk1_angles[2]);
96  remove_redundancies().apply(replacement);
97  if (conditional) {
98  circ.substitute_conditional(replacement, v, Circuit::VertexDeletion::No);
99  } else {
100  circ.substitute(replacement, v, Circuit::VertexDeletion::No);
101  }
102  circ.add_phase(tk1_angles[3]);
103  bin.push_back(v);
104  success = true;
105  }
106  circ.remove_vertices(
107  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
108  return success;
109 }

111 Transform rebase_tket() {
112  std::function<Circuit(const Expr&, const Expr&, const Expr&)> tk1_to_tk1 =
113  [](const Expr& alpha, const Expr& beta, const Expr& gamma) {
114  Circuit c(1);
115  c.add_op<unsigned>(OpType::TK1, {alpha, beta, gamma}, {0});
116  return c;
117  };
118  return rebase_factory({OpType::CX, OpType::TK1}, CircPool::CX(), tk1_to_tk1);
119 }
