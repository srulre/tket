22 bool Rewrite::decompose_boxes_fun(ZXDiagram& diag) {
23  std::list<ZXVert> to_decompose;
24  BGL_FORALL_VERTICES(v, *diag.graph, ZXGraph) {
25  if (diag.get_zxtype(v) == ZXType::ZXBox) to_decompose.push_back(v);
26  }
27  bool success = !to_decompose.empty();
28  while (!to_decompose.empty()) {
29  ZXVert box = to_decompose.front();
30  to_decompose.pop_front();
31  const ZXBox& zxb = diag.get_vertex_ZXGen<ZXBox>(box);
32  const ZXDiagram& inner = *zxb.get_diagram();
33  // Substitute diagram in place of box
34  auto iso = diag.copy_graph(inner, false);
35  std::map<Wire, Wire> incident_map;
36  for (const Wire& w : diag.adj_wires(box)) {
37  WireProperties wp = diag.get_wire_info(w);
38  ZXVert s = diag.source(w);
39  ZXVert t = diag.target(w);
40  if (s == box) {
41  ZXVert b = iso.first.at(inner.boundary.at(*wp.source_port));
42  Wire bw = diag.adj_wires(b).at(0);
43  if (diag.end_of(bw, b) == WireEnd::Source) {
44  s = diag.target(bw);
45  wp.source_port = wp.target_port;
46  } else {
47  s = diag.source(bw);
48  wp.source_port = wp.source_port;
49  }
50  }
51  if (t == box) {
52  ZXVert b = iso.first.at(inner.boundary.at(*wp.target_port));
53  Wire bw = diag.adj_wires(b).at(0);
54  if (diag.end_of(bw, b) == WireEnd::Source) {
55  t = diag.target(bw);
56  wp.target_port = wp.target_port;
57  } else {
58  t = diag.source(bw);
59  wp.target_port = wp.source_port;
60  }
61  }
62  Wire new_w = diag.add_wire(s, t, wp);
63  incident_map.emplace(w, new_w);
64  }
65  // Remove box and boundaries, and add new boxes to recursively decompose
66  diag.remove_vertex(box);
67  for (const std::pair<const ZXVert, ZXVert>& pair : iso.first) {
68  ZXType type = diag.get_zxtype(pair.second);
69  if (is_boundary_type(type))
70  diag.remove_vertex(pair.second);
71  else if (type == ZXType::ZXBox)
72  to_decompose.push_back(pair.second);
73  }
74  }
75  return success;
76 }

80 bool Rewrite::basic_wires_fun(ZXDiagram& diag) {
81  ZXGen_ptr qhad =
82  std::make_shared<const PhasedGen>(ZXType::Hbox, -1, QuantumType::Quantum);
83  ZXGen_ptr chad = std::make_shared<const PhasedGen>(
84  ZXType::Hbox, -1, QuantumType::Classical);
85  WireVec targets;
86  BGL_FORALL_EDGES(w, *diag.graph, ZXGraph) {
87  if (diag.get_wire_type(w) == ZXWireType::H) targets.push_back(w);
88  }
89  for (const Wire& w : targets) {
90  WireProperties wp = diag.get_wire_info(w);
91  ZXGen_ptr had = (wp.qtype == QuantumType::Quantum) ? qhad : chad;
92  ZXVert h = diag.add_vertex(had);
93  ZXVert s = diag.source(w);
94  ZXVert t = diag.target(w);
95  wp.type = ZXWireType::Basic;
96  WireProperties wp2 = wp;
97  wp.target_port = std::nullopt;
98  wp2.source_port = std::nullopt;
99  diag.add_wire(s, h, wp);
100  diag.add_wire(h, t, wp);
101  diag.remove_wire(w);
102  }
103  return !targets.empty();
104 }
