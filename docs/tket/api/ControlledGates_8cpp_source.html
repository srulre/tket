47 // `n` = size of incrementer, circuit returned is of size `n+1`
48 /* this is slightly less efficient than perhaps it could be -- asymptotically it
49  is still good. In an ideal world, this would decompose the incrementers
50  smarter for the "even" case */
51 Circuit incrementer_borrow_1_qubit(unsigned n) {
52  bool is_odd = n % 2;
53  Circuit circ(n + 1);
54  if (n < 6) {
55  if (n > 4)
56  circ.append_qubits(CircPool::C4X_normal_decomp(), {0, 1, 2, 3, 4});
57  if (n > 3) circ.append_qubits(CircPool::C3X_normal_decomp(), {0, 1, 2, 3});
58  if (n > 2) circ.add_op<unsigned>(OpType::CCX, {0, 1, 2});
59  if (n > 1) circ.add_op<unsigned>(OpType::CX, {0, 1});
60  if (n > 0) circ.add_op<unsigned>(OpType::X, {0});
61  return circ;
62  }
63  unsigned j;
64  unsigned k;
65  // j is bottom qubits, k is top qubits
66  // k + j = n + 1 (total no. of qbs)
67  if (is_odd) {
68  /* if the number of bits we are incrementing is odd, we can just split the
69  incrementer in 2 and use the `incrementer_borrow_n_qubits` method twice */
70  j = (n + 1) / 2, k = (n + 1) / 2;
71  } else {
72  /* otherwise, we will also have to pull out a cnx */
73  j = n / 2 + 1, k = n / 2;
74  }
75 
76  Circuit top_incrementer = incrementer_borrow_n_qubits(k);
77  std::vector<unsigned> top_qbs(2 * k);
78  for (unsigned i = 0; i != k; ++i) {
79  top_qbs[2 * i] = i + k; // borrowed qubits
80  top_qbs[2 * i + 1] = i; // qbs we are trying to increment
81  }
82 
83  Circuit cnx_top;
84  std::vector<unsigned> cnx1_qbs;
85  if (k == 3) { // code is unreachable if k<3
86  cnx_top = CircPool::C3X_normal_decomp();
87  cnx1_qbs = {0, 1, 2, n};
88  } else if (k == 4) {
89  cnx_top = CircPool::C4X_normal_decomp();
90  cnx1_qbs = {0, 1, 2, 3, n};
91  } else {
92  cnx_top = lemma72(k); // k controls on cnx
93  cnx1_qbs.resize(
94  2 * k - 2); // size of replacement using borrowed qbs = 2*k-1
95  std::iota(cnx1_qbs.begin(), cnx1_qbs.end(), 0);
96  cnx1_qbs.push_back(n); // target is last qubit
97  }
98 
99  Circuit bottom_incrementer;
100  std::vector<unsigned> bot_qbs;
101  if (is_odd) {
102  bottom_incrementer = incrementer_borrow_n_qubits(j);
103  bot_qbs.resize(2 * j);
104  for (unsigned i = 0; i != j; ++i) {
105  bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
106  if (i != 0)
107  bot_qbs[2 * i + 1] =
108  i + j -
109  1; // 3,5...n //other qbs we are actually trying to increment
110  }
111  bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
112  } else {
113  if (j == 4) { // code is unreachable if j<4
114  bottom_incrementer.add_blank_wires(4);
115  bottom_incrementer.append_qubits(
116  CircPool::C3X_normal_decomp(), {0, 1, 2, 3});
117  bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
118  bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
119  bottom_incrementer.add_op<unsigned>(OpType::X, {0});
120  bot_qbs = {n, n - 3, n - 2, n - 1};
121  } else if (j == 5) {
122  bottom_incrementer.add_blank_wires(5);
123  bottom_incrementer.append_qubits(
124  CircPool::C4X_normal_decomp(), {0, 1, 2, 3, 4});
125  bottom_incrementer.append_qubits(
126  CircPool::C3X_normal_decomp(), {0, 1, 2, 3});
127  bottom_incrementer.add_op<unsigned>(OpType::CCX, {0, 1, 2});
128  bottom_incrementer.add_op<unsigned>(OpType::CX, {0, 1});
129  bottom_incrementer.add_op<unsigned>(OpType::X, {0});
130  bot_qbs = {n, n - 4, n - 3, n - 2, n - 1};
131  } else {
132  // insert peeled-out cnx
133  Circuit cnx_bot = lemma72(j - 1);
134  std::vector<unsigned> cnx2_qbs(
135  2 * j - 3); // lemma 7.2 uses 2j-3 qubits for a (j-1)-controlled X
136  for (unsigned i = 0; i < j - 2; ++i) {
137  cnx2_qbs[i] = k + i;
138  }
139  cnx2_qbs[j - 2] = n; // replace the last control
140  for (unsigned i = 0; i < j - 3; ++i) {
141  cnx2_qbs[j + i - 1] = i;
142  }
143  cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
144 
145  circ.append_qubits(cnx_bot, cnx2_qbs);
146  bottom_incrementer = incrementer_borrow_n_qubits(
147  j - 1); // insert incrementer over remaining qubits
148  bot_qbs.resize(2 * j - 2);
149  for (unsigned i = 0; i != j - 1; ++i) {
150  bot_qbs[2 * i] = i; // 0,2,4...n-1 //borrowed qubits
151  if (i != 0)
152  bot_qbs[2 * i + 1] =
153  i + k -
154  1; // 3,5...n //other qbs we are actually trying to increment
155  }
156  bot_qbs[1] = n; // incremented qubit 0 in incrementer is bottom one
157  }
158  }
159 
160  circ.append_qubits(bottom_incrementer, bot_qbs);
161  circ.add_op<unsigned>(
162  OpType::X,
163  {n}); // to convert controlled-incrementer to larger incrementer
164  for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
165  circ.append_qubits(cnx_top, cnx1_qbs);
166  if (!is_odd && j > 5) {
167  // insert peeled-out cnx
168  Circuit cnx_bot = lemma72(j - 1);
169  std::vector<unsigned> cnx2_qbs(2 * j - 3);
170  for (unsigned i = 0; i < j - 1; ++i) {
171  cnx2_qbs[i] = k + i;
172  }
173  cnx2_qbs[j - 2] = n; // the last control
174  for (unsigned i = 0; i < j - 3; ++i) {
175  cnx2_qbs[j + i - 1] = i;
176  }
177  cnx2_qbs[2 * j - 4] = n - 1; // the target of the peeled out cnx
178  circ.append_qubits(cnx_bot, cnx2_qbs);
179  }
180  circ.append_qubits(bottom_incrementer, bot_qbs);
181  circ.add_op<unsigned>(OpType::X, {n});
182  circ.append_qubits(cnx_top, cnx1_qbs);
183  for (unsigned i = k; i != n; ++i) circ.add_op<unsigned>(OpType::CX, {n, i});
184  circ.append_qubits(top_incrementer, top_qbs);
185  return circ;
186 }

188 // an optimised version of
189 // https://algassert.com/circuits/2015/06/12/Constructing-Large-Increment-Gates.html
190 /* every second qubit (0,2,4...) is a borrowed qubit */
191 Circuit incrementer_borrow_n_qubits(unsigned n) {
192  const unsigned N = 2 * n;
193  Circuit circ(N);
194  /* deal with small cases where borrowing qubits is unnecessary */
195  if (n < 6) {
196  if (n > 4)
197  circ.append_qubits(CircPool::C4X_normal_decomp(), {1, 3, 5, 7, 9});
198  if (n > 3) circ.append_qubits(CircPool::C3X_normal_decomp(), {1, 3, 5, 7});
199  if (n > 2) circ.add_op<unsigned>(OpType::CCX, {1, 3, 5});
200  if (n > 1) circ.add_op<unsigned>(OpType::CX, {1, 3});
201  if (n > 0) circ.add_op<unsigned>(OpType::X, {1});
202  return circ;
203  }
204 
205  for (unsigned i = 1; i < N; ++i) {
206  if (i % 2)
207  circ.add_op<unsigned>(OpType::CX, {0, i});
208  else
209  circ.add_op<unsigned>(OpType::X, {i});
210  }
211 
212  circ.add_op<unsigned>(OpType::X, {N - 1});
213 
214  for (unsigned i = 2; i < N; ++(++i)) {
215  std::vector<unsigned> ladder_down_qbs = {i - 2, i - 1, i};
216  circ.append_qubits(CircPool::ladder_down(), ladder_down_qbs);
217  }
218  circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
219  for (unsigned i = N - 2; i > 1; --(--i)) {
220  std::vector<unsigned> tof_up_qbs = {i - 2, i - 1, i};
221  circ.add_op<unsigned>(OpType::CCX, tof_up_qbs);
222  }
223 
224  for (unsigned i = 2; i < N; ++(++i)) {
225  std::vector<unsigned> ladder_down_2_qbs = {i - 2, i - 1, i};
226  circ.append_qubits(CircPool::ladder_down_2(), ladder_down_2_qbs);
227  }
228  circ.add_op<unsigned>(OpType::CX, {N - 2, N - 1});
229  for (unsigned i = N - 2; i > 1; --(--i)) {
230  std::vector<unsigned> ladder_up_qbs = {i - 2, i - 1, i};
231  circ.append_qubits(CircPool::ladder_up(), ladder_up_qbs);
232  }
233  for (unsigned i = 1; i < N; ++(++i))
234  circ.add_op<unsigned>(OpType::CX, {0, i});
235  return circ;
236 }

238 // decompose CnX gate using
239 // https://algassert.com/circuits/2015/06/22/Using-Quantum-Gates-instead-of-Ancilla-Bits.html
240 // `n` = no. of controls
241 Circuit cnx_normal_decomp(unsigned n) {
242  /* handle low qb edge cases */
243  bool insert_c4xs; // dictate whether to add C4Xs or n>4 controlled Xs
244  // when bootstrapping
245  switch (n) {
246  case 0: {
247  return CircPool::X();
248  }
249  case 1: {
250  return CircPool::CX();
251  }
252  case 2: {
253  return CircPool::CCX_normal_decomp();
254  }
255  case 3: {
256  return CircPool::C3X_normal_decomp();
257  }
258  case 4: {
259  return CircPool::C4X_normal_decomp();
260  }
261  case 5: {
262  insert_c4xs = true;
263  break;
264  }
265  default: {
266  insert_c4xs = false;
267  break;
268  }
269  }
270 
271  Circuit circ(n + 1);
272  std::vector<unsigned> cnx_qbs(n - 1);
273  std::iota(cnx_qbs.begin(), cnx_qbs.end(), 0);
274  cnx_qbs.push_back(n);
275  // first, bootstrap an ancilla qubit
276  circ.add_op<unsigned>(OpType::H, {n});
277  Vertex cnx1;
278  if (insert_c4xs)
279  circ.append_qubits(CircPool::C4X_normal_decomp(), {cnx_qbs});
280  else {
281  cnx1 = circ.add_op<unsigned>(
282  OpType::CnX,
283  {cnx_qbs}); /* the CnXs can be decomposed using lemma 7.3 */
284  }
285  circ.add_op<unsigned>(OpType::Tdg, {n});
286  Vertex cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
287  if (!insert_c4xs) {
288  Edge e1 = circ.get_nth_in_edge(cx, 0);
289  lemma73(circ, {e1, cnx1}); // replace first CnX using lemma 7.3
290  }
291  circ.add_op<unsigned>(OpType::T, {n});
292  Vertex cnx2;
293  if (insert_c4xs)
294  circ.append_qubits(CircPool::C4X_normal_decomp(), {cnx_qbs});
295  else {
296  cnx2 = circ.add_op<unsigned>(OpType::CnX, {cnx_qbs});
297  }
298  circ.add_op<unsigned>(OpType::Tdg, {n});
299  cx = circ.add_op<unsigned>(OpType::CX, {n - 1, n});
300  Edge e2 = circ.get_nth_in_edge(cx, 0);
301  if (!insert_c4xs) lemma73(circ, {e2, cnx2});
302  circ.add_op<unsigned>(OpType::T, {n});
303  circ.add_op<unsigned>(OpType::H, {n});
304 
305  // add incremented shift pattern
306  Circuit incrementer = incrementer_borrow_1_qubit(n);
307  circ.append(incrementer);
308 
309  // z rotation layer #1
310  std::vector<Op_ptr> z_rots(n);
311  double angle = -0.25;
312  for (unsigned i = 0; i < n - 1; ++i) {
313  unsigned m = n - 1 - i;
314  z_rots[i] = get_op_ptr(OpType::Rz, angle);
315  circ.add_op<unsigned>(z_rots[i], {m});
316  angle /= 2;
317  }
318 
319  // decremented shift pattern
320  for (unsigned i = 0; i < n; ++i) {
321  circ.add_op<unsigned>(OpType::X, {i});
322  }
323  circ.append(incrementer);
324  for (unsigned i = 0; i < n; ++i) {
325  circ.add_op<unsigned>(OpType::X, {i});
326  }
327 
328  // z rotation layer #2
329  for (unsigned i = 0; i < n - 1; ++i) {
330  unsigned m = n - 1 - i;
331  Expr ang = z_rots[i]->get_params()[0];
332  circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {m});
333  }
334  Expr ang = z_rots[n - 2]->get_params()[0];
335  circ.add_op<unsigned>(get_op_ptr(OpType::Rz, -ang), {0});
336 
337  decomp_CCX().apply(circ);
338  circ.add_phase(std::pow(0.5, n + 1));
339  return circ;
340 }

342 /* assumes vert is controlled Ry with 1 control */
343 /* decomposes CRy into 2 CXs and 2 Ry gates */
344 static Circuit lemma54(const Expr& angle) {
345  Circuit new_circ(2);
346  std::vector<Expr> A_params = {angle / 2.};
347  std::vector<Expr> B_params = {-angle / 2.};
348  const Op_ptr A = get_op_ptr(OpType::Ry, A_params);
349  const Op_ptr B = get_op_ptr(OpType::Ry, B_params);
350  new_circ.add_op<unsigned>(A, {1});
351  new_circ.add_op<unsigned>(OpType::CX, {0, 1});
352  new_circ.add_op<unsigned>(B, {1});
353  new_circ.add_op<unsigned>(OpType::CX, {0, 1});
354  return new_circ;
355 }

357 // unsigned -> which column the change is in
358 static unsigned find_first_differing_val(
359  const std::deque<bool>& d1, const std::deque<bool>& d2) {
360  unsigned N = d1.size();
361  if (N != d2.size())
362  throw ControlDecompError(
363  "Error in `find_first_differing_val`: Deques are of differing "
364  "sizes");
365  for (unsigned i = 0; i < N; ++i) {
366  if (d1[i] != d2[i]) return i;
367  }
368  throw ControlDecompError(
369  "Error in `find_first_differing_val`: No change between deques");
370 }

372 // optimal decomposition of CnRy and CnZ for 2 < n < 8 according to 1995
373 // paper... can do better with ZH calculus?
374 static Circuit lemma71(
375  unsigned arity, const Expr& angle, const OpType& cr_type) {
376  unsigned m_controls = arity - 1;
377  if (m_controls < 2)
378  throw Unsupported(
379  "No point using Lemma 7.1 to decompose a gate with less than 2 "
380  "controls");
381  if (m_controls > 7)
382  throw Unsupported(
383  "Using Lemma 7.1 to decompose a gate with more than 7 controls "
384  "is inefficient");
385  if (cr_type != OpType::CRy && cr_type != OpType::CU1)
386  throw Unsupported(
387  "The implementation currently only supports CU1 and CRy ");
388 
389  GrayCode gc = gen_graycode(m_controls);
390  unsigned n_square_roots = m_controls - 1;
391 
392  Circuit rep(arity);
393  Expr param;
394  std::optional<double> reduced = eval_expr_mod(angle, 4);
395  if (reduced)
396  param = reduced.value();
397  else
398  param = angle;
399 
400  param = param / (1 << (n_square_roots));
401 
402  const Op_ptr V_op = get_op_ptr(cr_type, param);
403  const Op_ptr V_dg = get_op_ptr(cr_type, -param);
404 
405  unsigned control_qb = 0;
406  unsigned last = 0;
407  rep.add_op<unsigned>(V_op, {0, m_controls});
408  // we ignore the 0...0 term, and the first one is always trivial
409  // so start from 2
410  for (unsigned i = 2; i < gc.size(); ++i) {
411  unsigned change = find_first_differing_val(gc[i], gc[i - 1]);
412  for (unsigned j = 1; j < gc[i].size(); ++j) {
413  if (gc[i][j] == 1) last = j;
414  }
415  if (change < control_qb) {
416  rep.add_op<unsigned>(OpType::CX, {change, control_qb});
417  }
418  if (change == control_qb) {
419  throw ControlDecompError("Error in graycode iteration");
420  }
421  if (change > control_qb) {
422  rep.add_op<unsigned>(OpType::CX, {control_qb, change});
423  }
424 
425  if ((i % 2) == 0) {
426  rep.add_op<unsigned>(V_dg, {last, m_controls});
427  } else
428  rep.add_op<unsigned>(V_op, {last, m_controls});
429  control_qb = last;
430  }
431  unsigned correct_gate_count =
432  ((1 << m_controls) - 1) + ((1 << m_controls) - 2);
433  if (rep.n_gates() != correct_gate_count)
434  throw ControlDecompError("Error in Lemma 7.1: Gate count is incorrect");
435  auto [vit, vend] = boost::vertices(rep.dag);
436  VertexSet bin;
437  for (auto next = vit; vit != vend; vit = next) {
438  ++next;
439  Vertex v = *vit;
440  if (!bin.contains(v)) {
441  OpType optype = rep.get_OpType_from_Vertex(v);
442  if (optype == OpType::CRy || optype == OpType::CU1) {
443  Expr v_angle = rep.get_Op_ptr_from_Vertex(v)->get_params()[0];
444  Circuit replacement = (optype == OpType::CRy)
445  ? CircPool::CRy_using_CX(v_angle)
446  : CircPool::CU1_using_CX(v_angle);
447  Subcircuit sub{rep.get_in_edges(v), rep.get_all_out_edges(v), {v}};
448  rep.substitute(replacement, sub, Circuit::VertexDeletion::No);
449  bin.insert(v);
450  }
451  }
452  }
453  rep.remove_vertices(
454  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
455  return rep;
456 }

458 //`control_m` = number of controls
459 static Circuit lemma72(unsigned control_m) {
460  if (control_m < 3)
461  throw Unsupported(
462  "Cannot decompose a gate with " + std::to_string(control_m) +
463  " controls using Lemma 7.2");
464  const unsigned n = 2 * control_m - 1;
465 
466  Circuit ccx_circ(n);
467  unsigned diff = n - control_m;
468  for (unsigned i = control_m - 1; i > 1; --i) {
469  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
470  }
471  ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
472  for (unsigned i = 2; i < control_m; ++i) {
473  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
474  }
475  for (unsigned i = control_m - 2; i > 1; --i) {
476  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
477  }
478  ccx_circ.add_op<unsigned>(OpType::CCX, {0, 1, control_m});
479  for (unsigned i = 2; i < control_m - 1; ++i) {
480  ccx_circ.add_op<unsigned>(OpType::CCX, {i, i + diff - 1, i + diff});
481  }
482  if (ccx_circ.count_gates(OpType::CCX) != 4 * (control_m - 2))
483  throw ControlDecompError("Error in Lemma 7.2: CCX gate count is incorrect");
484  return ccx_circ;
485 }

487 /* this is specifically for performing corollary 7.4 via lemma 7.3 & lemma 7.2 -
488  * optimal decomp of a CnX gate */
489 /* for corollary 7.4, n >= 7 */
490 // this is a decomposition of a CnX gate using one dirty ancilla
491 static void lemma73(Circuit& circ, const std::pair<Edge, Vertex>& pairy) {
492  Vertex original_cnx = pairy.second;
493  Edge original_spare_edge = pairy.first;
494  EdgeVec in_edges = circ.get_in_edges(original_cnx);
495  const unsigned N =
496  in_edges.size() + 1; // number of qubits in new circuit to replace
497  if (N < 5)
498  throw Unsupported(
499  "Lemma 7.3 cannot decompose CnX with n = " + std::to_string(N - 1));
500 
501  EdgeVec out_edges = circ.get_all_out_edges(original_cnx);
502 
503  in_edges.insert(in_edges.begin() + in_edges.size() - 1, original_spare_edge);
504  out_edges.insert(
505  out_edges.begin() + out_edges.size() - 1, original_spare_edge);
506 
507  Subcircuit to_delete{in_edges, out_edges, {original_cnx}};
508  bool odd_N = N % 2;
509  unsigned m1 = (N + 1) / 2; // number of controls on the first type of CnX
510  unsigned m2 = N - m1 - 1; // number of controls on the second type of CnX
511 
512  // make new circuit to substitute later
513  Circuit new_circ(N);
514  const Op_ptr cnx_op1 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m1 + 1);
515  const Op_ptr cnx_op2 = get_op_ptr(OpType::CnX, std::vector<Expr>{}, m2 + 1);
516  std::vector<unsigned> qbs_m1(m1 + 1);
517  std::iota(qbs_m1.begin(), qbs_m1.begin() + m1, 0);
518  qbs_m1[m1] = N - 1;
519  std::vector<unsigned> qbs_m2(m2 + 1);
520  std::iota(qbs_m2.begin(), qbs_m2.end(), N - (m2 + 1));
521 
522  // add ladder of CnXs to the correct qubits
523  Vertex a = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
524  Vertex b = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
525  Vertex c = new_circ.add_op<unsigned>(cnx_op1, qbs_m1);
526  Vertex d = new_circ.add_op<unsigned>(cnx_op2, qbs_m2);
527 
528  /* first, replace vertex a, putting circuit at back of new_circ */
529  unsigned cutsize = odd_N ? N : N - 1;
530  EdgeVec cut1(cutsize);
531  VertexVec out_verts = new_circ.q_outputs();
532  if (odd_N) {
533  for (unsigned i = 0; i < N - 2; ++i)
534  cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
535  cut1[N - 2] = new_circ.get_nth_in_edge(out_verts[N - 1], 0);
536  cut1[N - 1] = new_circ.get_nth_in_edge(out_verts[N - 2], 0);
537  } else {
538  for (unsigned i = 0; i < cutsize; ++i)
539  cut1[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
540  }
541 
542  Circuit a_replacement;
543  if (m1 == 1) {
544  a_replacement = CircPool::CX();
545  } else if (m1 == 2) {
546  a_replacement = CircPool::CCX();
547  } else
548  a_replacement = lemma72(m1); // returns circuit of size 2*m1 - 1
549  new_circ.cut_insert(a_replacement, cut1);
550  new_circ.remove_vertex(
551  a, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
552 
553  VertexSet normal_decomp_vertices;
554 
555  Circuit b_replacement;
556  if (m2 == 1) {
557  b_replacement = CircPool::CX();
558  } else if (m2 == 2) {
559  b_replacement = CircPool::CCX();
560  } else
561  b_replacement = lemma72(m2); // returns circuit of size 2*m2 - 1
562

564 
565  // reassign cut to back of circuit
566  EdgeVec frontier(N);
567  for (unsigned i = 0; i < N; ++i)
568  frontier[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
569 
570  EdgeVec cut2(b_qubits);
571  for (unsigned i = N - m2 - 1; i < N - 1; ++i)
572  cut2[i - (N - m2 - 1)] =
573  frontier[i]; // N-1 - (N-m2-1) = m2 (all the controls)
574  for (unsigned i = 0; i < b_qubits - (m2 + 1); ++i)
575  cut2[i + m2] = frontier[i]; // empty wires
576  cut2[b_qubits - 1] = frontier[N - 1]; // target
577 
578  new_circ.cut_insert(b_replacement, cut2);
579  new_circ.remove_vertex(
580  b, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
581  Vertex last_path = out_verts[N - 1];
582  Edge init_back_edge = new_circ.get_nth_in_edge(last_path, 0);
583  Vertex init_back_path = new_circ.source(init_back_edge);
584  normal_decomp_vertices.insert(init_back_path);
585  init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
586  init_back_path = new_circ.source(init_back_edge);
587  OpType backstop = new_circ.get_OpType_from_Vertex(init_back_path);
588  while ((backstop != OpType::CCX) && !is_initial_q_type(backstop)) {
589  init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
590  init_back_path = new_circ.source(init_back_edge);
591  backstop = new_circ.get_OpType_from_Vertex(init_back_path);
592  }
593  normal_decomp_vertices.insert(init_back_path);
594  /* now, replace vertex c */
595  EdgeVec cut3(cutsize);
596  if (odd_N) {
597  for (unsigned i = 0; i < N - 2; ++i)
598  cut3[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
599  cut3[N - 2] = new_circ.get_nth_in_edge(out_verts[N - 1], 0);
600  cut3[N - 1] = new_circ.get_nth_in_edge(out_verts[N - 2], 0);
601  } else
602  for (unsigned i = 0; i < cutsize; ++i)
603  cut3[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
604 
605  new_circ.cut_insert(a_replacement, cut3);
606  new_circ.remove_vertex(
607  c, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
608 
609  /* now, replace vertex d */
610  for (unsigned i = 0; i < N; ++i)
611  frontier[i] = new_circ.get_nth_in_edge(out_verts[i], 0);
612  EdgeVec cut4(b_qubits);
613  for (unsigned i = N - m2 - 1; i < N - 1; ++i)
614  cut4[i - (N - m2 - 1)] =
615  frontier[i]; // N-1 - (N-m2-1) = m2 (all the controls)
616  for (unsigned i = 0; i < b_qubits - (m2 + 1); ++i)
617  cut4[i + m2] = frontier[i]; // empty wires
618  cut4[b_qubits - 1] = frontier[N - 1]; // target
619 
620  new_circ.cut_insert(b_replacement, cut4);
621  new_circ.remove_vertex(
622  d, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::Yes);
623 
624  last_path = out_verts[N - 1];
625  init_back_edge = new_circ.get_nth_in_edge(last_path, 0);
626  init_back_path = new_circ.source(init_back_edge);
627  normal_decomp_vertices.insert(init_back_path);
628  init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
629  init_back_path = new_circ.source(init_back_edge);
630  backstop = new_circ.get_OpType_from_Vertex(init_back_path);
631  while ((backstop != OpType::CCX) && !is_initial_q_type(backstop)) {
632  init_back_edge = new_circ.get_last_edge(init_back_path, init_back_edge);
633  init_back_path = new_circ.source(init_back_edge);
634  backstop = new_circ.get_OpType_from_Vertex(init_back_path);
635  }
636  normal_decomp_vertices.insert(init_back_path);
637 
638  if (m1 > 2 && m2 > 2 && (new_circ.count_gates(OpType::CCX) != 8 * N - 40))
639  throw ControlDecompError("Error in Lemma 7.3: CCX gate count is incorrect");
640  /* now, replace CCXs with either CX circuit modulo phase shift or just typical
641  * Toffoli decomposition */
642 
643  auto [vit, vend] = boost::vertices(new_circ.dag);
644  for (auto next = vit; vit != vend; vit = next) {
645  ++next;
646  if (new_circ.get_OpType_from_Vertex(*vit) == OpType::CCX) {

649  new_circ.get_all_out_edges(*vit),
650  {*vit}};

652  new_circ.substitute(
653  CircPool::CCX_modulo_phase_shift(), sub,
654  Circuit::VertexDeletion::Yes);
655  } else {
656  new_circ.substitute(
657  CircPool::CCX_normal_decomp(), sub, Circuit::VertexDeletion::Yes);
658  }
659  }
660  }
661  if (m1 > 2 && m2 > 2 && new_circ.count_gates(OpType::CX) != 24 * N - 108)
662  throw ControlDecompError("Error in Lemma 7.3: CX gate count is incorrect");
663 
664  circ.substitute(new_circ, to_delete, Circuit::VertexDeletion::Yes);
665 }

667 // N must be >= 3
668 static void lemma79(
669  Circuit& replacement, unsigned N, const Expr& angle,
670  candidate_t& CCX_candidates) {
671  replacement.add_blank_wires(N);
672 
673  std::vector<Expr> A_params = {angle / 2.};
674  std::vector<Expr> B_params = {-angle / 2.};
675  const Op_ptr A = get_op_ptr(OpType::CnRy, A_params, 2);
676  const Op_ptr B = get_op_ptr(OpType::CnRy, B_params, 2);
677 
678  Vertex vA = replacement.add_op<unsigned>(A, {N - 2, N - 1}); // A
679  std::vector<unsigned> cnx_qbs(N - 1);
680  std::iota(cnx_qbs.begin(), --cnx_qbs.end(), 0);
681  cnx_qbs[N - 2] = N - 1;
682  const Op_ptr cnx = get_op_ptr(OpType::CnX, std::vector<Expr>{}, N - 1);
683  Vertex firstCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
684  Vertex vB = replacement.add_op<unsigned>(B, {N - 2, N - 1}); // B
685  CCX_candidates.push_back(
686  {boost::edge(vA, vB, replacement.dag).first, firstCnX});
687  Vertex secondCnX = replacement.add_op<unsigned>(cnx, cnx_qbs);
688  Edge out_edge_spare = replacement.get_nth_out_edge(vB, 0);
689  CCX_candidates.push_back({out_edge_spare, secondCnX});
690 }
