102 static bool convert_to_zyz(Circuit &circ) {
103  static const Expr half = SymEngine::div(Expr(1), Expr(2));
104  bool success = decompose_single_qubits_TK1().apply(circ);
105  VertexList bin;
106  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
107  if (circ.n_in_edges(v) != 1) continue;
108  const Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
109  if (op->get_type() == OpType::TK1) {
110  std::vector<Expr> params = op->get_params();
111  Circuit replacement(1);
112  Expr a = params[2] + half;
113  Expr b = params[1];
114  Expr c = params[0] - half;
115  if (!equiv_0(a, 4)) {
116  replacement.add_op<unsigned>(OpType::Rz, a, {0});
117  }
118  if (!equiv_0(b, 4)) {
119  replacement.add_op<unsigned>(OpType::Ry, b, {0});
120  }
121  if (!equiv_0(c, 4)) {
122  replacement.add_op<unsigned>(OpType::Rz, c, {0});
123  }
124  Subcircuit sub = {
125  {circ.get_in_edges(v)}, {circ.get_all_out_edges(v)}, {v}};
126  bin.push_back(v);
127  circ.substitute(replacement, sub, Circuit::VertexDeletion::No);
128  success = true;
129  }
130  }
131  circ.remove_vertices(
132  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
133  return success;
134 }

197 Transform decompose_ZYZ_to_TK1() {
198  return Transform([](Circuit &circ) {
199  bool success = false;
200  static const Expr zero(0);
201  static const Expr half = SymEngine::div(Expr(1), Expr(2));
202  VertexList bin;
203  VertexVec inputs = circ.q_inputs();
204  for (VertexVec::iterator i = inputs.begin(); i != inputs.end(); ++i) {
205  Edge e = circ.get_nth_out_edge(*i, 0);
206  Vertex v = circ.target(e);
207  while (!is_final_q_type(circ.get_OpType_from_Vertex(v))) {
208  if (circ.get_OpType_from_Vertex(v) == OpType::Rz) {
209  const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
210  Expr angle_1 = v_g->get_params()[0];
211  Edge e1 = circ.get_next_edge(v, e);
212  Vertex v2 = circ.target(e1);
213  if (circ.get_OpType_from_Vertex(v2) == OpType::Ry) {
214  const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
215  Expr angle_2 = v2_g->get_params()[0];
216  Edge e2 = circ.get_next_edge(v2, e1);
217  Vertex v3 = circ.target(e2);
218  bin.push_back(v2);
219  circ.remove_vertex(
220  v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
221  Expr angle_3 = zero;
222  if (circ.get_OpType_from_Vertex(v3) == OpType::Rz) {
223  const Op_ptr v3_g = circ.get_Op_ptr_from_Vertex(v3);
224  angle_3 = v3_g->get_params()[0];
225  circ.remove_vertex(
226  v3, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
227  bin.push_back(v3);
228  }
229  std::vector<Expr> new_params = {
230  angle_3 + half, angle_2, angle_1 - half};
231  circ.dag[v] = {get_op_ptr(OpType::TK1, new_params)};
232  } else {
233  circ.dag[v] = {get_op_ptr(OpType::TK1, {zero, zero, angle_1})};
234  }
235  } else if (circ.get_OpType_from_Vertex(v) == OpType::Ry) {
236  const Op_ptr v_g = circ.get_Op_ptr_from_Vertex(v);
237  Expr angle_2 = v_g->get_params()[0];
238  Expr angle_3 = zero;
239  Edge e1 = circ.get_next_edge(v, e);
240  Vertex v2 = circ.target(e1);
241  if (circ.get_OpType_from_Vertex(v2) == OpType::Rz) {
242  const Op_ptr v2_g = circ.get_Op_ptr_from_Vertex(v2);
243  angle_3 = v2_g->get_params()[0];
244  circ.remove_vertex(
245  v2, Circuit::GraphRewiring::Yes, Circuit::VertexDeletion::No);
246  bin.push_back(v2);
247  }
248  std::vector<Expr> new_params = {angle_3 + half, angle_2, -half};
249  circ.dag[v] = {get_op_ptr(OpType::TK1, new_params)};
250  }
251  e = circ.get_next_edge(v, e);
252  v = circ.target(e);
253  }
254  }
255  circ.remove_vertices(
256  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
257  return success;
258  });
259 }

324 /* --Rz(a)--Rx(b)--R(c)-- => --Rz(a+c)--PhasedX(b,c)-- */
325 Transform decompose_ZX_to_HQS1() {
326  return Transform([](Circuit &circ) {
327  bool success = false;
328  VertexList to_bin;
329  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
330  if (circ.get_OpType_from_Vertex(v) == OpType::Rx) {
331  success = true;
332  const Op_ptr g = circ.get_Op_ptr_from_Vertex(v);
333  Expr theta = g->get_params()[0];
334  Vertex prev_vert = *circ.get_predecessors(v).begin();
335  Vertex next_vert = *circ.get_successors(v).begin();
336  if (circ.get_OpType_from_Vertex(prev_vert) == OpType::Rz &&
337  circ.get_OpType_from_Vertex(next_vert) == OpType::Rz) {
338  const Op_ptr prev_g = circ.get_Op_ptr_from_Vertex(prev_vert);
339  const Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next_vert);
340  Expr phi = next_g->get_params()[0];
341  std::vector<Expr> params{theta, phi};
342  circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
343  circ.remove_vertex(
344  next_vert, Circuit::GraphRewiring::Yes,
345  Circuit::VertexDeletion::No);
346  to_bin.push_back(next_vert);
347  Expr new_param = prev_g->get_params()[0] + phi;
348  circ.dag[prev_vert].op = get_op_ptr(OpType::Rz, new_param);
349  } else {
350  // if no Rz, initialise a PhasedX op with theta=Rx.params[0],phi=0
351  Expr phi(0);
352  std::vector<Expr> params{theta, phi};
353  circ.dag[v].op = get_op_ptr(OpType::PhasedX, params);
354  }
355  }
356  }
357  circ.remove_vertices(
358  to_bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
359  remove_redundancies().apply(circ);
360  return success;
361  });
362 }

364 // Decompose CX into MolmerSorensen as:
365 // ---C--- -V-S-|-H-
366 // | --> XX(pi/4)
367 // ---X--- -----|-Vdg-
368 Transform decompose_MolmerSorensen() {
369  return Transform([](Circuit &circ) {
370  bool success = false;
371  VertexList bin;
372  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
373  if (circ.get_OpType_from_Vertex(v) != OpType::CX) continue;
374  EdgeVec outs = circ.get_all_out_edges(v);
375  if (outs.size() == 2) {
376  Vertex next = circ.target(outs[0]);
377  // Is the next operation equivalent to an Rx, up to phase?
378  Op_ptr next_g = circ.get_Op_ptr_from_Vertex(next);
379  OpType next_type = next_g->get_type();
380  if (is_single_qubit_type(next_type) && !is_projective_type(next_type)) {
381  std::vector<Expr> angles = as_gate_ptr(next_g)->get_tk1_angles();
382  if (equiv_0(angles[0], 2) && equiv_0(angles[2], 2)) {
383  Expr angle = angles[1];
384  Expr phase = angles[3];
385  if (!equiv_0(angles[0], 4)) phase += 1;
386  if (!equiv_0(angles[2], 4)) phase += 1;
387  Edge next_e = circ.get_nth_out_edge(next, 0);
388  Vertex last = circ.target(next_e);
389  if (circ.get_OpType_from_Vertex(last) == OpType::CX &&
390  circ.get_nth_in_edge(last, 1) == outs[1]) {
391  // Recognise exp(-i XX * angle * pi/2)
392  const Op_ptr op_ptr = get_op_ptr(OpType::XXPhase, angle);
393  circ.dag[v] = {op_ptr};
394  bin.push_back(next);
395  circ.remove_vertex(
396  next, Circuit::GraphRewiring::Yes,
397  Circuit::VertexDeletion::No);
398  bin.push_back(last);
399  circ.remove_vertex(
400  last, Circuit::GraphRewiring::Yes,
401  Circuit::VertexDeletion::No);
402  circ.add_phase(phase);
403  success = true;
404  continue;
405  }
406  }
407  }
408  // Replace remaining CX gates
409  Subcircuit sub = {{circ.get_in_edges(v)}, {outs}, {v}};
410  bin.push_back(v);
411  circ.substitute(
412  CircPool::CX_using_XXPhase_1(), sub, Circuit::VertexDeletion::No);
413  success = true;
414  }
415  }
416  circ.remove_vertices(
417  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
418  return success;
419  });
420 }

568 static Circuit clifford_from_tk1(int i, int j, int k) {
569  std_cliff_spec_t spec = tk1_table[i % 4][j % 4][k % 4];
570  if (i >= 4) spec.p += 1;
571  if (j >= 4) spec.p += 1;
572  if (k >= 4) spec.p += 1;
573 
574  Circuit c(1);
575  if (spec.Z0) {
576  c.add_op<unsigned>(OpType::Z, {0});
577  }
578  if (spec.X1) {
579  c.add_op<unsigned>(OpType::X, {0});
580  }
581  if (spec.S2) {
582  c.add_op<unsigned>(OpType::S, {0});

1066 // Any PhasedX or NPhasedX gate is replaced by an NPhasedX that is global
1067 Transform globalise_PhasedX(bool squash) {
1068  // The key bit: choose the decomposition strategy depending on the current
1069  // beta angles.
1070  //
1071  // Given the set of PhasedX gates to synthesise, choose which decomposition
1072  // strategy. Valid strategies are 0, 1, 2, corresponding to the number of
1073  // NPhasedX gates to be inserted.
1074  //
1075  // The current strategy is rather simple: it chooses to insert a single
1076  // NPhasedX whenever it would solve the current problem and there are further
1077  // PhasedX left (meaning that the rest of the computation can be deferred till
1078  // later), otherwise inserts 2x PhasedX.
1079  auto choose_strategy = [](const PhasedXFrontier &frontier, unsigned n_target,
1080  unsigned n_all) {
1081  if (n_all == 0 || n_target == 0) {
1082  return 0;
1083  }
1084  if (n_target == 1 && n_all == 1) {
1085  return 1;
1086  }
1087  if (n_target == 1 && frontier.are_phasedx_left()) {
1088  return 1;
1089  }
1090  return 2;
1091  };
1092 
1093  // the actual transform
1094  return Transform([squash, choose_strategy](Circuit &circ) {
1095  // if we squash, we start by removing all NPhasedX gates
1096  if (squash) {
1097  Transforms::decompose_NPhasedX().apply(circ);
1098  }
1099 
1100  std::vector<unsigned> range_qbs(circ.n_qubits());
1101  std::iota(range_qbs.begin(), range_qbs.end(), 0);
1102  PhasedXFrontier frontier(circ);
1103 
1104  // find a total ordering of boundary gates (aka non-NPhasedX multi-qb gates)
1105  auto is_boundary = [&circ](Vertex v) {
1106  Op_ptr op = circ.get_Op_ptr_from_Vertex(v);
1107  return PhasedXFrontier::is_interval_boundary(op);
1108  };
1109  // get a lexicographic ordering of boundary vertices
1110  auto vertices_in_order = circ.vertices_in_order();
1111  auto r = vertices_in_order | boost::adaptors::filtered(is_boundary);
1112  OptVertexVec boundary_gates(r.begin(), r.end());
1113  // Add sentinel to process the gates after last boundary_gate.
1114  // This is required to force flush the frontier after the last multi-qubit
1115  // gate.
1116  OptVertex optv;
1117  boundary_gates.push_back(optv);
1118 
1119  // whether transform is successful (always true if squash=true)
1120  bool success = squash;
1121 
1122  // Loop through each multi-qb gate.
1123  // At each iteration, decide how to decompose the single-qb gates
1124  // immediately preceding the current multi-qb gate into global gates.
1125  //
1126  // The rationale: defer any computation until just before the next multi-qb
1127  // gate. At that point, the single-qb gates on the qubits where the multi-qb
1128  // gate is acting HAVE TO be decomposed. Figure out how to do that and take
1129  // care of the garbage you might have created on other qubits at a later
1130  // point.
1131 
1132  for (OptVertex v : boundary_gates) {
1133  // the qubits whose intervals must be decomposed into global gates
1134  std::set<unsigned> curr_qubits;
1135  while (true) {
1136  if (v) {
1137  curr_qubits = frontier.qubits_ending_in(*v);
1138  } else {
1139  curr_qubits.clear();
1140  for (unsigned i = 0; i < circ.n_qubits(); ++i) {
1141  curr_qubits.insert(curr_qubits.end(), i);
1142  }
1143  }
1144  if (squash) {
1145  frontier.squash_intervals();
