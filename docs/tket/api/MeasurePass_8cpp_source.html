23 Transform delay_measures() {
24  return Transform([](Circuit &circ) {
25  bool success = false;
26  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
27  if (circ.get_OpType_from_Vertex(v) == OpType::Measure) {
28  Edge c_out_edge = circ.get_nth_out_edge(v, 1);
29  if (!circ.detect_final_Op(circ.target(c_out_edge)) ||
30  circ.n_out_edges_of_type(v, EdgeType::Boolean) != 0) {
31  throw CircuitInvalidity(
32  "Cannot commute Measure through classical operations to the end "
33  "of the circuit");
34  }
35  Edge out_edge = circ.get_nth_out_edge(v, 0);
36  Edge current_edge = out_edge;
37  Vertex current_vertex = circ.target(current_edge);
38  port_t current_port = circ.get_target_port(current_edge);
39  Op_ptr current_op = circ.get_Op_ptr_from_Vertex(current_vertex);
40  OpType current_optype = current_op->get_type();
41  while (!is_final_q_type(current_optype) &&
42  (current_optype == OpType::SWAP ||
43  current_op->commutes_with_basis(Pauli::Z, current_port))) {
44  if (current_optype == OpType::SWAP) {
45  // Update from SWAP
46  current_edge =
47  circ.get_nth_out_edge(current_vertex, 1 - current_port);
48  } else {
49  // Update to successor
50  current_edge = circ.get_nth_out_edge(current_vertex, current_port);
51  }
52  current_vertex = circ.target(current_edge);
53  current_port = circ.get_target_port(current_edge);
54  current_op = circ.get_Op_ptr_from_Vertex(current_vertex);
55  current_optype = current_op->get_type();
56  }
57  // If we haven't moved it to an output, we can't continue
58  if (!is_final_q_type(current_optype)) {
59  throw CircuitInvalidity(
60  "Cannot commute Measure through quantum gates to the end of the "
61  "circuit");
62  }
63  // If the measure was already at an output, nothing to do
64  if (current_edge == out_edge) continue;
65  Edge in_edge = circ.get_nth_in_edge(v, 0);
66  // Rewire measure
67  circ.add_edge(
68  {circ.source(in_edge), circ.get_source_port(in_edge)},
69  {circ.target(out_edge), circ.get_target_port(out_edge)},
70  EdgeType::Quantum);
71  circ.remove_edge(in_edge);
72  circ.remove_edge(out_edge);
73  circ.add_edge(
74  {circ.source(current_edge), circ.get_source_port(current_edge)},
75  {v, 0}, EdgeType::Quantum);
76  circ.add_edge({v, 0}, {current_vertex, 0}, EdgeType::Quantum);
77  circ.remove_edge(current_edge);
78  success = true;
79  }
80  }
81  return success;
82  });
83 }
