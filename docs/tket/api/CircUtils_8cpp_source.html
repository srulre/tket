35 Eigen::Matrix2cd get_matrix(const Circuit &circ, const Vertex &vert) {
36  const Op_ptr op = circ.get_Op_ptr_from_Vertex(vert);
37  if (op->get_type() != OpType::TK1) {
38  throw NotImplemented(
39  "Cannot obtain matrix from gate: " + op->get_name() +
40  ". Try rebasing to tket's internal representation.");
41  }
42  std::vector<Expr> ps = op->get_params();
43  ps.push_back(0);
44  return get_matrix_from_tk1_angles(ps);
45 }

47 Eigen::Matrix2cd get_matrix_from_circ(const Circuit &circ) {
48  if (circ.n_qubits() != 1)
49  throw CircuitInvalidity(
50  "Getting Matrix: expected 1 qubit circuit, found " +
51  std::to_string(circ.n_qubits()));
52  Complex factor = std::exp(i_ * PI * eval_expr(circ.get_phase()).value());
53  VertexVec qpath = circ.qubit_path_vertices(circ.all_qubits()[0]);
54  unsigned N = qpath.size();
55  if (N == 2) return factor * Eigen::Matrix2cd::Identity();
56  Eigen::Matrix2cd m = get_matrix(circ, qpath[N - 2]);
57  for (unsigned x = N - 3; x >= 1; --x) {
58  m = m * get_matrix(circ, qpath[x]);
59  }
60  return factor * m;
61 }

63 Eigen::Matrix4cd get_matrix_from_2qb_circ(const Circuit &circ) {
64  std::vector<QPathDetailed> all_paths = circ.all_qubit_paths();
65  std::map<Vertex, Eigen::Matrix4cd> v_to_op;
66  Eigen::Matrix4cd cnot, tonc, swap;
67  // clang-format off
68  cnot << 1, 0, 0, 0,
69  0, 1, 0, 0,
70  0, 0, 0, 1,
71  0, 0, 1, 0;
72  tonc << 1, 0, 0, 0,
73  0, 0, 0, 1,
74  0, 0, 1, 0,
75  0, 1, 0, 0;
76  swap << 1, 0, 0, 0,
77  0, 0, 1, 0,
78  0, 1, 0, 0,
79  0, 0, 0, 1;
80  // clang-format on
81 
82  for (unsigned uqb = 0; uqb < 2; uqb++) {
83  for (QPathDetailed::iterator it = all_paths[uqb].begin();
84  it != all_paths[uqb].end(); ++it) {
85  const Op_ptr o = circ.get_Op_ptr_from_Vertex(it->first);
86  switch (o->get_type()) {
87  case OpType::Input:
88  case OpType::Create:
89  case OpType::Output:
90  case OpType::Discard: {
91  v_to_op[it->first] = Eigen::Matrix4cd::Identity();
92  break;
93  }
94  case OpType::SWAP: {
95  if (uqb == 0) v_to_op[it->first] = swap;
96  break;
97  }
98  case OpType::CX: {
99  if (uqb == 0) {
100  if (it->second == 0) {
101  v_to_op[it->first] = cnot;
102  } else {
103  v_to_op[it->first] = tonc;
104  }
105  }
106  break;
107  }
108  case OpType::TK2: {
109  auto params = o->get_params();
110  TKET_ASSERT(params.size() == 3);
111  v_to_op[it->first] = get_matrix_from_2qb_circ(
112  CircPool::TK2_using_CX(params[0], params[1], params[2]));
113  break;
114  }
115  default: {
116  if (o->get_desc().is_gate() && circ.n_in_edges(it->first) == 1 &&
117  circ.n_out_edges(it->first) == 1) {
118  const Op_ptr g = o;
119  std::vector<Expr> ps = as_gate_ptr(g)->get_tk1_angles();
120  Eigen::Matrix2cd mat = get_matrix_from_tk1_angles(ps);
121  if (uqb == 0) {
122  v_to_op[it->first] =
123  Eigen::kroneckerProduct(mat, Eigen::Matrix2cd::Identity());
124  } else {
125  v_to_op[it->first] =
126  Eigen::kroneckerProduct(Eigen::Matrix2cd::Identity(), mat);
127  }
128  } else
129  throw NotImplemented(
130  "Cannot obtain matrix from op " + o->get_name());
131  }
132  }
133  }
134  }
135  Eigen::Matrix4cd m = Eigen::Matrix4cd::Identity();
136  SliceVec slices = circ.get_slices();
137  for (const Slice &s : slices) {
138  for (const Vertex &v : s) {
139  m = v_to_op[v] * m;
140  }
141  }
142  return std::exp(i_ * PI * eval_expr(circ.get_phase()).value()) * m;
143 }

145 Circuit two_qubit_canonical(const Eigen::Matrix4cd &U) {
146  if (!is_unitary(U)) {
147  throw std::invalid_argument(
148  "Non-unitary matrix passed to two_qubit_canonical");
149  }
150 
151  auto [K1, A, K2] = get_information_content(U);
152 
153  K1 /= pow(K1.determinant(), 0.25);
154  K2 /= pow(K2.determinant(), 0.25);
155  auto [a, b, c] = A;
156 
157  // Decompose single qubits
158  auto [K1a, K1b] = kronecker_decomposition(K1);
159  auto [K2a, K2b] = kronecker_decomposition(K2);
160 
161  Circuit result(2);
162 
163  std::vector<double> angles_q0 = tk1_angles_from_unitary(K2a);
164  std::vector<double> angles_q1 = tk1_angles_from_unitary(K2b);
165  result.add_op<unsigned>(
166  OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
167  result.add_op<unsigned>(
168  OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
169 
170  result.add_op<unsigned>(OpType::TK2, {a, b, c}, {0, 1});
171 
172  angles_q0 = tk1_angles_from_unitary(K1a);
173  angles_q1 = tk1_angles_from_unitary(K1b);
174  result.add_op<unsigned>(
175  OpType::TK1, {angles_q0.begin(), angles_q0.end() - 1}, {0});
176  result.add_op<unsigned>(
177  OpType::TK1, {angles_q1.begin(), angles_q1.end() - 1}, {1});
178 
179  // this fixes phase if decomposition is exact
180  Eigen::Matrix4cd reminder = get_matrix_from_2qb_circ(result).adjoint() * U;
181  const Complex phase = reminder(0, 0); // reminder = phase * I
182  result.add_phase(arg(phase) / PI);
183  return result;
184 }

186 // Factorize U as VD where V corresponds to a 2-CX circuit and
187 // D = diag(z, z*, z*, z). Return V and z.
188 static std::pair<Eigen::Matrix4cd, Complex> decompose_VD(
189  const Eigen::Matrix4cd &U) {
190  if (!is_unitary(U)) {
191  throw std::invalid_argument("Non-unitary matrix passed to decompose_VD");
192  }
193 
194  // The calculations below are derived from the proof of Proposition V.2 in
195  // https://arxiv.org/abs/quant-ph/0308033.
196 
197  Eigen::Matrix4cd u = U / pow(U.determinant(), 0.25);
198  Complex a = u(3, 0) * u(0, 3) - u(2, 0) * u(1, 3) - u(1, 0) * u(2, 3) +
199  u(0, 0) * u(3, 3);
200  Complex b = u(3, 1) * u(0, 2) - u(2, 1) * u(1, 2) - u(1, 1) * u(2, 2) +
201  u(0, 1) * u(3, 2);
202  // Now we want to find z such that |z|=1 and (az* - bz) is real.
203  // The numerical stability of this function is a concern when a is close to
204  // -b*. This problem can be demonstrated in artificially constructed
205  // examples (passing unitaries very close to, but not quite, the identity to
206  // the functions below). In these cases the product VD (or DV) may not
207  // approximate U to within Eigen's default tolerance. Is there a way to
208  // dodge this issue?
209  Complex w = a + std::conj(b);
210  double d = std::abs(w);
211  // If w = 0 then we can set z = 1.
212  Complex z = (d < EPS) ? 1. : w / d;
213  Complex z0 = sqrt(z);
214  Complex z1 = std::conj(z0);
215  Eigen::Matrix4cd V = U;
216  V.col(0) *= z1;
217  V.col(1) *= z0;
218  V.col(2) *= z0;
219  V.col(3) *= z1;
220  return {V, z0};
221 }

223 static void replace_TK2_2CX(Circuit &circ) {
224  VertexList bin;
225  BGL_FORALL_VERTICES(v, circ.dag, DAG) {
226  if (circ.get_OpType_from_Vertex(v) != OpType::TK2) continue;
227  auto params = circ.get_Op_ptr_from_Vertex(v)->get_params();
228  TKET_ASSERT(params.size() == 3);
229  TKET_ASSERT(equiv_0(params[2], 4));
230  Circuit sub = CircPool::approx_TK2_using_2xCX(params[0], params[1]);
231  bin.push_back(v);
232  circ.substitute(sub, v, Circuit::VertexDeletion::No);
233  }
234  TKET_ASSERT(bin.size() == 1);
235  circ.remove_vertices(
236  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
237 }

239 std::pair<Circuit, Complex> decompose_2cx_VD(const Eigen::Matrix4cd &U) {
240  auto [V, z0] = decompose_VD(U);
241  Circuit circ = two_qubit_canonical(V);
242  replace_TK2_2CX(circ);
243  return {circ, z0};
244 }

246 std::pair<Circuit, Complex> decompose_2cx_DV(const Eigen::Matrix4cd &U) {
247  auto [V, z0] = decompose_VD(U.adjoint());
248  V.adjointInPlace();
249  Circuit circ = two_qubit_canonical(V);
250  replace_TK2_2CX(circ);
251  return {circ, std::conj(z0)};
252 }

254 Circuit phase_gadget(unsigned n_qubits, const Expr &t, CXConfigType cx_config) {
255  // Handle n_qubits==0 as a special case, or the calculations below
256  // go badly wrong.
257  Circuit new_circ(n_qubits);
258  if (n_qubits == 0) {
259  new_circ.add_phase(-t / 2);
260  return new_circ;
261  }
262  switch (cx_config) {
263  case CXConfigType::Snake: {
264  for (unsigned i = n_qubits - 1; i != 0; --i) {
265  unsigned j = i - 1;
266  new_circ.add_op<unsigned>(OpType::CX, {i, j});
267  }
268  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
269  for (unsigned i = 0; i != n_qubits - 1; ++i) {
270  unsigned j = i + 1;
271  new_circ.add_op<unsigned>(OpType::CX, {j, i});
272  }
273  break;
274  }
275  case CXConfigType::Star: {
276  for (unsigned i = n_qubits - 1; i != 0; --i) {
277  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
278  }
279  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
280  for (unsigned i = 1; i != n_qubits; ++i) {
281  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
282  }
283  break;
284  }
285  case CXConfigType::Tree: {
286  unsigned complete_layers = floor(log2(n_qubits));
287  unsigned dense_end = pow(2, complete_layers);
288  for (unsigned i = 0; i < n_qubits - dense_end; i++)
289  new_circ.add_op<unsigned>(
290  OpType::CX, {dense_end + i, dense_end - 1 - i});
291  for (unsigned step_size = 1; step_size < dense_end; step_size *= 2) {
292  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
293  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
294  }
295  new_circ.add_op<unsigned>(OpType::Rz, t, {0});
296  for (unsigned step_size = dense_end / 2; step_size >= 1; step_size /= 2) {
297  for (unsigned i = 0; i < dense_end; i += 2 * step_size)
298  new_circ.add_op<unsigned>(OpType::CX, {i + step_size, i});
299  }
300  for (unsigned i = 0; i < n_qubits - dense_end; i++)
301  new_circ.add_op<unsigned>(
302  OpType::CX, {dense_end + i, dense_end - 1 - i});
303  break;
304  }
305  case CXConfigType::MultiQGate: {
306  std::vector<std::vector<unsigned>> conjugations;
307  int sign_correction = 1;
308  for (int q = n_qubits - 1; q > 0; q -= 2) {
309  if (q - 1 > 0) {
310  unsigned i = q, j = q - 1;
311  // this is only equal to the CX decompositions above
312  // up to phase, but phase differences are cancelled out by
313  // its dagger XXPhase(-1/2) below.
314  new_circ.add_op<unsigned>(OpType::H, {i});
315  new_circ.add_op<unsigned>(OpType::H, {j});
316  new_circ.add_op<unsigned>(OpType::XXPhase3, 0.5, {i, j, 0});
317  sign_correction *= -1;
318  conjugations.push_back({i, j, 0});
319  } else {
320  unsigned i = q;
321  new_circ.add_op<unsigned>(OpType::CX, {i, 0});
322  conjugations.push_back({i, 0});
323  }
324  }
325  new_circ.add_op<unsigned>(OpType::Rz, sign_correction * t, {0});
326  for (const auto &conj : conjugations) {
327  if (conj.size() == 2) {
328  new_circ.add_op<unsigned>(OpType::CX, conj);
329  } else {
330  TKET_ASSERT(conj.size() == 3);
331  new_circ.add_op<unsigned>(OpType::XXPhase3, -0.5, conj);
332  new_circ.add_op<unsigned>(OpType::H, {conj[0]});
333  new_circ.add_op<unsigned>(OpType::H, {conj[1]});
334  }
335  }
336  break;
337  }
338  }
339  return new_circ;
340 }

342 Circuit pauli_gadget(
343  const std::vector<Pauli> &paulis, const Expr &t, CXConfigType cx_config) {
344  unsigned n = paulis.size();
345  Circuit circ(n);
346  std::vector<unsigned> qubits;
347  for (unsigned i = 0; i < n; i++) {
348  switch (paulis[i]) {
349  case Pauli::I:
350  break;
351  case Pauli::X:
352  circ.add_op<unsigned>(OpType::H, {i});
353  qubits.push_back(i);
354  break;
355  case Pauli::Y:
356  circ.add_op<unsigned>(OpType::V, {i});
357  qubits.push_back(i);
358  break;
359  case Pauli::Z:
360  qubits.push_back(i);
361  break;
362  }
363  }
364  if (qubits.empty()) {
365  circ.add_phase(-t / 2);
366  } else {
367  Vertex v = circ.add_op<unsigned>(OpType::PhaseGadget, t, qubits);
368  Circuit cx_gadget = phase_gadget(circ.n_in_edges(v), t, cx_config);
369  Subcircuit sub = {circ.get_in_edges(v), circ.get_all_out_edges(v), {v}};
370  circ.substitute(cx_gadget, sub, Circuit::VertexDeletion::Yes);
371  for (unsigned i = 0; i < n; i++) {
372  switch (paulis[i]) {
373  case Pauli::I:
374  break;
375  case Pauli::X:
376  circ.add_op<unsigned>(OpType::H, {i});
377  break;
378  case Pauli::Y:
379  circ.add_op<unsigned>(OpType::Vdg, {i});
380  break;
381  case Pauli::Z:
382  break;
383  }
384  }
385  }
386  return circ;
387 }

394 Circuit with_TK2(Gate_ptr op) {
395  std::vector<Expr> params = op->get_params();
396  unsigned n = op->n_qubits();
397  if (n == 0) {
398  return Circuit();
399  } else if (n == 1) {
400  Circuit c(1);
401  c.add_op(op, std::vector<unsigned>{0});
402  return c;
403  } else if (n == 2 && op->free_symbols().empty()) {
404  Eigen::Matrix4cd U = op->get_unitary();
405  auto [K1, A, K2] = get_information_content(U);
406  // Decompose single qubits
407  auto [K1a, K1b] = kronecker_decomposition(K1);
408  auto [K2a, K2b] = kronecker_decomposition(K2);
409  Circuit c(2);
410  std::vector<double> angles_K1a = tk1_angles_from_unitary(K1a);
411  std::vector<double> angles_K1b = tk1_angles_from_unitary(K1b);
412  std::vector<double> angles_K2a = tk1_angles_from_unitary(K2a);
413  std::vector<double> angles_K2b = tk1_angles_from_unitary(K2b);
414  c.add_op<unsigned>(
415  OpType::TK1, {angles_K2a.begin(), angles_K2a.end() - 1}, {0});
416  c.add_op<unsigned>(
417  OpType::TK1, {angles_K2b.begin(), angles_K2b.end() - 1}, {1});
418  double alpha = std::get<0>(A);
419  double beta = std::get<1>(A);
420  double gamma = std::get<2>(A);
421  c.add_op<unsigned>(OpType::TK2, {alpha, beta, gamma}, {0, 1});
422  c.add_op<unsigned>(
423  OpType::TK1, {angles_K1a.begin(), angles_K1a.end() - 1}, {0});
424  c.add_op<unsigned>(
425  OpType::TK1, {angles_K1b.begin(), angles_K1b.end() - 1}, {1});
426 
427  // Correct phase by computing the unitary and comparing with U:
428  Eigen::Matrix4cd V_K1 = Eigen::KroneckerProduct(
429  get_matrix_from_tk1_angles(
430  {angles_K1a[0], angles_K1a[1], angles_K1a[2], 0}),
431  get_matrix_from_tk1_angles(
432  {angles_K1b[0], angles_K1b[1], angles_K1b[2], 0}));
433  Eigen::Matrix4cd V_A =
434  internal::GateUnitaryMatrixImplementations::TK2(alpha, beta, gamma);
435  Eigen::Matrix4cd V_K2 = Eigen::KroneckerProduct(
436  get_matrix_from_tk1_angles(
437  {angles_K2a[0], angles_K2a[1], angles_K2a[2], 0}),
438  get_matrix_from_tk1_angles(
439  {angles_K2b[0], angles_K2b[1], angles_K2b[2], 0}));
440  Eigen::Matrix4cd V = V_K1 * V_A * V_K2;
441  Eigen::Matrix4cd R = V.adjoint() * U;
442  const Complex phase = R(0, 0); // R = phase * I
443  c.add_phase(arg(phase) / PI);
444 
445  return c;
446  }
447  // Now the non-trivial cases.
448  switch (op->get_type()) {
449  case OpType::ISWAP:
450  return CircPool::ISWAP_using_TK2(params[0]);
451  case OpType::PhasedISWAP:
452  return CircPool::PhasedISWAP_using_TK2(params[0], params[1]);
453  case OpType::XXPhase:
454  return CircPool::XXPhase_using_TK2(params[0]);
455  case OpType::YYPhase:
456  return CircPool::YYPhase_using_TK2(params[0]);
457  case OpType::ZZPhase:
458  return CircPool::ZZPhase_using_TK2(params[0]);
459  case OpType::NPhasedX:
460  return CircPool::NPhasedX_using_PhasedX(n, params[0], params[1]);
461  case OpType::ESWAP:
462  return CircPool::ESWAP_using_TK2(params[0]);
463  case OpType::FSim:
464  return CircPool::FSim_using_TK2(params[0], params[1]);
465  case OpType::CRx:
466  return CircPool::CRx_using_TK2(params[0]);
467  case OpType::CRy:
468  return CircPool::CRy_using_TK2(params[0]);
469  case OpType::CRz:
470  return CircPool::CRz_using_TK2(params[0]);
471  case OpType::CU1:
472  return CircPool::CU1_using_TK2(params[0]);
473  case OpType::XXPhase3:
474  return CircPool::XXPhase3_using_TK2(params[0]);
475  case OpType::CCX:
476  case OpType::CSWAP:
477  case OpType::BRIDGE:
478  case OpType::CU3:
479  case OpType::PhaseGadget: {
480  // As a first, inefficient, solution, decompose these into CX and then
481  // replace each CX with a TK2 (and some single-qubit gates).
482  // TODO Find more efficient decompositions for these gates.
483  Circuit c = with_CX(op);
484  replace_CX_with_TK2(c);
485  return c;
486  }
487  default:
488  throw CircuitInvalidity("Cannot decompose " + op->get_name());
489  }
490 }

492 Circuit with_CX(Gate_ptr op) {
493  OpType optype = op->get_type();
494  std::vector<Expr> params = op->get_params();
495  unsigned n = op->n_qubits();
496  if (n == 0) {
497  return Circuit();
498  } else if (n == 1) {
499  Circuit c(1);
500  c.add_op(op, std::vector<unsigned>{0});
501  return c;
502  }
503  switch (optype) {
504  case OpType::CX: {
505  Circuit c(2);
506  c.add_op(op, std::vector<unsigned>{0, 1});
507  return c;
508  }
509  case OpType::CCX:
510  return CircPool::CCX_normal_decomp();
511  case OpType::CY:
512  return CircPool::CY_using_CX();
513  case OpType::CZ:
514  return CircPool::CZ_using_CX();
515  case OpType::CH:
516  return CircPool::CH_using_CX();
517  case OpType::CV:
518  return CircPool::CV_using_CX();
519  case OpType::CVdg:
520  return CircPool::CVdg_using_CX();
521  case OpType::CSX:
522  return CircPool::CSX_using_CX();
523  case OpType::CSXdg:
524  return CircPool::CSXdg_using_CX();
525  case OpType::CRz:
526  return CircPool::CRz_using_CX(params[0]);
527  case OpType::CRx:
528  return CircPool::CRx_using_CX(params[0]);
529  case OpType::CRy:
530  return CircPool::CRy_using_CX(params[0]);
531  case OpType::CU1:
532  return CircPool::CU1_using_CX(params[0]);
533  case OpType::CU3:
534  return CircPool::CU3_using_CX(params[0], params[1], params[2]);
535  case OpType::PhaseGadget:
536  return phase_gadget(n, params[0], CXConfigType::Snake);
537  case OpType::SWAP:
538  return CircPool::SWAP_using_CX_0();
539  case OpType::CSWAP:
540  return CircPool::CSWAP_using_CX();
541  case OpType::BRIDGE:
542  return CircPool::BRIDGE_using_CX_0();
543  case OpType::ECR:
544  return CircPool::ECR_using_CX();
545  case OpType::ISWAP:

547  case OpType::ZZMax:

582 static Circuit CnU1(unsigned n_controls, Expr lambda) {
583  // CnU1(x) decomposes recursively as:
584  // C{n-1}U1(x/2)[ctrls]; U1(x/2)[tgt]; CnX; U1(-x/2)[tgt]; CnX
585  // We don't actually use recursion; just iterate starting with the first U1:
586  Circuit c(n_controls + 1);
587  Expr x = lambda / (1u << n_controls);
588  c.add_op<unsigned>(OpType::U1, x, {0});
589  std::vector<unsigned> cnx_qbs = {0};
590  for (unsigned i = 0; i < n_controls; i++) {
591  cnx_qbs.push_back(i + 1);
592  c.add_op<unsigned>(OpType::U1, x, {i + 1});
593  c.add_op<unsigned>(CNXTYPE(i + 2), cnx_qbs);
594  c.add_op<unsigned>(OpType::U1, -x, {i + 1});
595  c.add_op<unsigned>(CNXTYPE(i + 2), cnx_qbs);
596  x *= 2;
597  }
598  return c;
599 }

601 Circuit with_controls(const Circuit &c, unsigned n_controls) {
602  if (c.n_bits() != 0 || !c.is_simple()) {
603  throw CircuitInvalidity("Only default qubit register allowed");
604  }
605  if (c.has_implicit_wireswaps()) {
606  throw CircuitInvalidity("Circuit has implicit wireswaps");
607  }
608 
609  // Dispose of the trivial case
610  if (n_controls == 0) {
611  return c;
612  }
613 
614  unsigned c_n_qubits = c.n_qubits();
615 
616  // 1. Rebase to {CX, CCX, CnX, CnRy} and single-qubit gates
617  Circuit c1(c);
618  VertexList bin;
619  BGL_FORALL_VERTICES(v, c1.dag, DAG) {
620  Op_ptr op = c1.get_Op_ptr_from_Vertex(v);
621  OpType optype = op->get_type();
622  if (is_gate_type(optype)) {
623  if (is_projective_type(optype)) {
624  throw CircuitInvalidity("Projective operations present");
625  }
626  if (is_box_type(optype)) {
627  throw CircuitInvalidity("Undecomposed boxes present");
628  }
629  if (is_single_qubit_type(optype)) {
630  continue;
631  }
632  if (optype == OpType::CX || optype == OpType::CCX ||
633  optype == OpType::CnX || optype == OpType::CnRy) {
634  continue;
635  }
636  Circuit replacement = with_CX(as_gate_ptr(op));
637  c1.substitute(replacement, v, Circuit::VertexDeletion::No);
638  bin.push_back(v);
639  }
640  }
641  c1.remove_vertices(
642  bin, Circuit::GraphRewiring::No, Circuit::VertexDeletion::Yes);
643 
644  // Capture the phase. We may adjust this during replacements below.
645  Expr a = c1.get_phase();
646 
647  // 2. Replace all gates with controlled versions
648  Circuit c2(n_controls + c_n_qubits);
649  for (Circuit::CommandIterator cit = c1.begin(); cit != c1.end(); ++cit) {
650  Op_ptr op = cit->get_op_ptr();
651  unit_vector_t args = cit->get_args();
652  unsigned n_args = args.size();
653  unsigned n_new_args = n_controls + n_args;
654  qubit_vector_t new_args(n_new_args);
655  for (unsigned i = 0; i < n_controls; i++) {
656  new_args[i] = Qubit(i);
657  }
658  for (unsigned i = 0; i < n_args; i++) {
659  new_args[n_controls + i] = Qubit(n_controls + args[i].index()[0]);
660  }
661  OpType optype = op->get_type();
662  std::vector<Expr> params = op->get_params();
663  switch (optype) {
664  case OpType::noop:
665  break;
666  case OpType::X:
667  case OpType::CX:
668  case OpType::CCX:
669  case OpType::CnX:
670  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
671  break;
672  case OpType::Ry:
673  case OpType::CnRy:
674  c2.add_op<Qubit>(OpType::CnRy, params, new_args);
675  break;
676  default: {
677  std::vector<Expr> tk1_angles = as_gate_ptr(op)->get_tk1_angles();
678  Expr theta = tk1_angles[1];
679  Expr phi = tk1_angles[0] - 0.5;
680  Expr lambda = tk1_angles[2] + 0.5;
681  Expr t = tk1_angles[3] - 0.5 * (tk1_angles[0] + tk1_angles[2]);
682  // Operation is U3(theta, phi, lambda) + phase t.
683  // First absorb t in the overall phase.
684  a += t;
685  // Construct a multi-controlled U3, by extending the standard
686  // CU3-to-CX decomposition.
687  Qubit target = new_args[n_controls];
688  Circuit cnu1 = CnU1(n_controls - 1, 0.5 * (lambda + phi));
689  c2.append(cnu1);
690  c2.add_op<Qubit>(OpType::U1, 0.5 * (lambda - phi), {target});
691  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
692  c2.add_op<Qubit>(
693  OpType::U3, {-0.5 * theta, 0, -0.5 * (lambda + phi)}, {target});
694  c2.add_op<Qubit>(CNXTYPE(n_new_args), new_args);
695  c2.add_op<Qubit>(OpType::U3, {0.5 * theta, phi, 0}, {target});
696  } break;
697  }
698  }
699 
700  // 3. Account for phase by appending a CnU1 to the control qubits.
701  if (!equiv_0(a)) {
702  Circuit cnu1 = CnU1(n_controls - 1, a);
703  c2.append(cnu1);
704  }
705 
706  return c2;
707 }
