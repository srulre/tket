108 void fill_partial_mapping(
109  const qubit_vector_t &current_qubits, qubit_mapping_t &partial_mapping) {
110  unsigned up_nu = 0;
111  for (Qubit q : current_qubits) {
112  if (partial_mapping.find(q) == partial_mapping.end()) {
113  partial_mapping.insert({q, Node(Placement::unplaced_reg(), up_nu)});
114  up_nu++;
115  }
116  }
117 
118  if (std::any_of(
119  partial_mapping.begin(), partial_mapping.end(),
120  [&current_qubits](auto x) {
121  return std::find(
122  current_qubits.begin(), current_qubits.end(), x.first) ==
123  current_qubits.end();
124  })) {
125  tket_log()->warn("mapping contains qubits not present in circuit");
126  }
127 }

161 std::vector<qubit_mapping_t> NaivePlacement::get_all_placement_maps(
162  const Circuit &circ_) const {
163  qubit_mapping_t placement;
164  qubit_vector_t to_place;
165  std::vector<Node> placed;
166 
167  // Find which/if any qubits need placing
168  for (const Qubit &q : circ_.all_qubits()) {
169  Node n(q);
170  if (!this->arc_.node_exists(n)) {
171  to_place.push_back(n);
172  } else {
173  placed.push_back(n);
174  // if already placed, make sure qubit retains placement
175  placement.insert({n, n});
176  }
177  }
178  // avoid doing std::set_difference unless qubits need to be placed
179  unsigned n_placed = to_place.size();
180  if (n_placed > 0) {
181  std::vector<Node> difference,
182  architecture_nodes = this->arc_.get_all_nodes_vec();
183  std::set_difference(
184  architecture_nodes.begin(), architecture_nodes.end(), placed.begin(),
185  placed.end(), std::inserter(difference, difference.begin()));
186  // should always be enough remaining qubits to assign unplaced qubits to
187  TKET_ASSERT(difference.size() >= n_placed);
188  for (unsigned i = 0; i < n_placed; i++) {
189  // naively assign each qubit to some free node
190  placement.insert({to_place[i], difference[i]});
191  }
192  }
193  return {placement};
194 }

221 std::vector<qubit_mapping_t> GraphPlacement::get_all_placement_maps(
222  const Circuit &circ_) const {
223  QubitGraph q_graph = monomorph_interaction_graph(
224  circ_, arc_.n_connections(), config_.depth_limit);
225  std::vector<qubit_bimap_t> all_bimaps = monomorphism_edge_break(
226  arc_, q_graph, config_.vf2_max_matches, config_.timeout);
227  std::vector<qubit_mapping_t> all_qmaps;
228  qubit_vector_t all_qbs = circ_.all_qubits();
229  for (qubit_bimap_t bm : all_bimaps) {
230  qubit_mapping_t qm = bimap_to_map(bm.left);
231  fill_partial_mapping(all_qbs, qm);
232  all_qmaps.push_back(qm);
233  }
234  return all_qmaps;
235 }

242 std::vector<qubit_mapping_t> NoiseAwarePlacement::get_all_placement_maps(
243  const Circuit &circ_) const {
244  // Set up default placements configuation
245  // Place according to configuration
246  Monomorpher placer(circ_, arc_, characterisation_, config_);
247  std::vector<MapCost> results = placer.place(config_.depth_limit * 2);
248  std::sort(results.begin(), results.end());
249  std::vector<qubit_mapping_t> output;
250  qubit_vector_t all_qbs = circ_.all_qubits();
251  for (const MapCost &map_c : results) {
252  qubit_mapping_t map_ = map_c.map;
253  fill_partial_mapping(all_qbs, map_);
254  output.push_back(map_c.map);
255  }
256  return output;
257 }
